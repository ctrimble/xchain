32  /**
33   * Configure Hibernate.

35   * Run an XChain.

37   * @author John Trimble

42  public class ExecuteMojo extends AbstractXChainMojo {

44  	private static final String LIFE_CYCLE_CLASS_NAME = "org.xchain.framework.lifecycle.Lifecycle";
45  	private static final String HIBERNATE_LIFE_CYCLE_CLASS_NAME = "org.xchain.framework.hibernate.HibernateLifecycle";
46  	private static final String CONFIGURATION_CLASS_NAME = Configuration.class.getName();
47  	private static final String CONFIGURATION_CONFIGURE_CONFIGURATION_METHOD = "configure";
48  	private static final Class<?>[] CONFIGURATION_CONFIGURE_CONFIGURATION_ARGUMENTS = new Class<?>[] {File.class};
49  	private static final String CONFIGURATION_SET_CONFIGURATION_METHOD = "setConfiguration";
50  	private static final String CATALOG_FACTORY_CLASS_NAME = "org.xchain.framework.factory.CatalogFactory";
51  	private static final String CATALOG_CLASS_NAME = "org.xchain.Catalog";
52  	private static final String JXPATH_CONTEXT_CLASS_NAME = "org.apache.commons.jxpath.JXPathContext";
53  	private static final String LIFE_CYCLE_START_LIFE_CYCLE_METHOD = "startLifecycle";
54  	private static final String LIFE_CYCLE_STOP_LIFE_CYCLE_METHOD = "stopLifecycle";
55  	private static final String COMMAND_CLASS_NAME = "org.xchain.Command";
56  	private static final String CATALOG_FACTORY_INSTANCE_METHOD_NAME = "getInstance";
57  	private static final String CATALOG_FACTORY_CATALOG_METHOD_NAME = "getCatalog";
58  	private static final Class<?>[] CATALOG_FACTORY_CATALOG_ARGUMENT_TYPES = new Class<?>[] {String.class};
59  	private static final String CATALOG_COMMAND_METHOD_NAME = "getCommand";
60  	private static final Class<?>[] CATALOG_COMMAND_ARGUMENT_TYPES = new Class<?>[] {String.class};
61  	private static final String COMMAND_EXECUTE_METHOD_NAME = "execute";

96  	public void execute() throws MojoExecutionException, MojoFailureException {
97  		ClassLoader chainClassLoader = createClassLoader(Thread.currentThread().getContextClassLoader(), this.classpathElements);
98  		
99  		try {
100 			Class<?> lifeCycleClass = chainClassLoader.loadClass(LIFE_CYCLE_CLASS_NAME);
101 			Class<?> hibernateLifeCycleClass = chainClassLoader.loadClass(HIBERNATE_LIFE_CYCLE_CLASS_NAME);
102 			Class<?> catalogFactoryClass = chainClassLoader.loadClass(CATALOG_FACTORY_CLASS_NAME);
103 			Class<?> catalogClass = chainClassLoader.loadClass(CATALOG_CLASS_NAME);
104 			Class<?> jxpathContextClass = chainClassLoader.loadClass(JXPATH_CONTEXT_CLASS_NAME);
105 			Class<?> scopedJXPathContextClass = chainClassLoader.loadClass("org.xchain.framework.jxpath.ScopedJXPathContextImpl");
106 			Class<?> contextFactoryClass = chainClassLoader.loadClass("org.xchain.framework.jxpath.JXPathContextFactoryImpl");
107 			Class<?> scopedQNameVariablesClass = chainClassLoader.loadClass("org.xchain.framework.jxpath.ScopedQNameVariables");
108 			Class<?> commandClass = chainClassLoader.loadClass(COMMAND_CLASS_NAME);
109 			Class<?> configurationClass = chainClassLoader.loadClass(CONFIGURATION_CLASS_NAME);
110 			
111 			// LifeCycle methods
112 			Method startLifeCycleMethod = lifeCycleClass.getMethod(LIFE_CYCLE_START_LIFE_CYCLE_METHOD, new Class<?>[]{});
113 			Method stopLifeCycleMethod = lifeCycleClass.getMethod(LIFE_CYCLE_STOP_LIFE_CYCLE_METHOD, new Class<?>[]{});
114 			
115 			// CatalogFactory methods
116 			Method getInstanceMethod = catalogFactoryClass.getMethod(CATALOG_FACTORY_INSTANCE_METHOD_NAME, new Class<?>[]{});
117 			Method getCatalogMethod = catalogFactoryClass.getMethod(CATALOG_FACTORY_CATALOG_METHOD_NAME, CATALOG_FACTORY_CATALOG_ARGUMENT_TYPES);
118 			
119 			// Catalog methods
120 			Method getCommandMethod = catalogClass.getMethod(CATALOG_COMMAND_METHOD_NAME, CATALOG_COMMAND_ARGUMENT_TYPES);
121 			
122 			// Command methods
123 			Method executeMethod = commandClass.getMethod(COMMAND_EXECUTE_METHOD_NAME, jxpathContextClass);
124 			
125 			// JXPathContext methods
126 			Method newContextMethod = contextFactoryClass.getMethod("newContext", jxpathContextClass, Object.class);
127 			Method declareVariableMethod = scopedQNameVariablesClass.getMethod("declareVariable", String.class, Object.class);
128 			Method getVariablesMethod = scopedJXPathContextClass.getMethod("getVariables", new Class[] {});
129 			
130 			// Configuration methods
131 			Method configureMethod = configurationClass.getMethod(CONFIGURATION_CONFIGURE_CONFIGURATION_METHOD, CONFIGURATION_CONFIGURE_CONFIGURATION_ARGUMENTS);
132 			Method setConfigurationMethod = hibernateLifeCycleClass.getMethod(CONFIGURATION_SET_CONFIGURATION_METHOD, configurationClass);
133 			
134 			// Step 1: Switch Class loader
135 			ClassLoader originalContextClassLoader = Thread.currentThread().getContextClassLoader();
136 			Thread.currentThread().setContextClassLoader(chainClassLoader);
137 			
138 			// Step 2: Configure Lifecycle
139 			Object configurationObject = null;
140 			if( configuration != null && configuration.exists() ) {
141   			configurationObject = configurationClass.newInstance();
142   			configurationObject = configureMethod.invoke(configurationObject, this.configuration);
143   			setConfigurationMethod.invoke(null, new Object[]{configurationObject});
144 			}
145 			
146 			// Step 3: Start life cycle
147 			if( this.runXChainLifeCycle ) startLifeCycleMethod.invoke(null, new Object[]{});
148 			
149 			// Step 4: Get the catalog
150 			Object catalogFactoryObject = getInstanceMethod.invoke(null, new Object[] {});
151 			Object catalogObject = getCatalogMethod.invoke(catalogFactoryObject, new Object[] {this.catalog});
152 			
153 			// Step 5: Get the command
154 			Object commandObject = getCommandMethod.invoke(catalogObject, new Object[] {this.dispatcherCommand});
155 			
156 			// Step 6: Configure the context
157 			Object jxpathContextFactoryObject = contextFactoryClass.newInstance();
158 			Object jxpathContextObject = newContextMethod.invoke(jxpathContextFactoryObject, null, new Object());
159 			Object variablesObject = getVariablesMethod.invoke(jxpathContextObject, new Object[] {});
160 			declareVariableMethod.invoke(variablesObject, "command", this.command);
161 			declareVariableMethod.invoke(variablesObject, "exportSchema", this.exportSchema);
162 			declareVariableMethod.invoke(variablesObject, "configuration", configurationObject);
163 
164 			// Step 7: Execute the command
165 			executeMethod.invoke(commandObject, jxpathContextObject);
166 			
167 			// Step 8: Stop life cycle
168 			if( this.runXChainLifeCycle ) stopLifeCycleMethod.invoke(null, new Object[]{});
169 			
170 			// Step 9: Switch Class loader back
171 			Thread.currentThread().setContextClassLoader(originalContextClassLoader);
172 			
173 		} catch (ClassNotFoundException e) {
174 			e.printStackTrace();
175 		} catch (SecurityException e) {
176 			e.printStackTrace();
177 		} catch (NoSuchMethodException e) {
178 			e.printStackTrace();
179 		} catch (IllegalArgumentException e) {
180 			e.printStackTrace();
181 		} catch (IllegalAccessException e) {
182 			e.printStackTrace();
183 		} catch (InvocationTargetException e) {
184 			e.printStackTrace();
185 		} catch (InstantiationException e) {
186 		  e.printStackTrace();
187 		}
188 	}

190 	private ClassLoader createClassLoader(ClassLoader parent, Collection<String> classPathElementFilePaths) {
191 		IsolatedClassLoader isolatedClassLoader = new IsolatedClassLoader(parent);
192 		if( classPathElementFilePaths != null && classPathElementFilePaths.size() > 0 ) {
193 			for( String classElementString : classPathElementFilePaths ) {
194 				File classElementFile = new File(classElementString);
195 				try {
196 					URL url = classElementFile.toURL();
197 					isolatedClassLoader.addURL(url);
198 				} catch (MalformedURLException e) {
199 					e.printStackTrace();
200 				}
201 			}
202 		}
203 		
204 		return isolatedClassLoader;
205 	}
