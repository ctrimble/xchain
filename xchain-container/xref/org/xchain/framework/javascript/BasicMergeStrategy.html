34  /**
35   * Port of Chris's original code from <code>org.xchain.namespaces.javascript.MergeJavascript</code> for merging 
36   * multiple JavaScript files together. 
37   * 
38   * @author Christian Trimble
39   * @author John Trimble
40   * @author Josh Kennedy
41   */
42  public class BasicMergeStrategy implements IMergeStrategy {
43    private static Logger log = LoggerFactory.getLogger(BasicMergeStrategy.class);
44    
45    public void merge(String manifestSystemId, OutputStream output) throws Exception {
46      // parse the manifest and output the files as we encounter them.
47      String jsSystemId = null;
48      String absoluteJsSystemId = null;
49      URL manifestUrl = null;
50      URL jsUrl = null;
51      InputStream manifestIn = null;
52      InputStream jsIn = null;
53      OutputStream servletOut = null;
54      LineNumberReader reader = null;
55      PrintWriter servletWriter = null;
56      try {
57        servletWriter = new PrintWriter(new OutputStreamWriter(servletOut = output));
58  
59        servletWriter.append("//\n// Manifest System Id: ").append(manifestSystemId).append("\n//\n").flush();
60  
61        // get the url for the absolute system id.
62        manifestUrl = UrlFactory.getInstance().newUrl(manifestSystemId);
63  
64        // create a reader that will allow us to track the line number of the system id we are working with.
65        reader = new LineNumberReader(new InputStreamReader(manifestIn = manifestUrl.openStream()));
66  
67        while( (jsSystemId = reader.readLine()) != null ) {
68          absoluteJsSystemId = URI.create(manifestSystemId).resolve(jsSystemId).toString();
69          servletWriter.append("//\n// Script System Id: ").append(absoluteJsSystemId).append("\n//\n").flush();
70          try {
71            jsUrl = UrlFactory.getInstance().newUrl(absoluteJsSystemId);
72            jsIn = jsUrl.openStream();
73            copyStream(jsIn, servletOut, 2048);
74          }
75          catch( Exception e ) {
76            if( log.isErrorEnabled() ) {
77              log.error("Could not copy the contents of '"+absoluteJsSystemId+" to the servlet output stream.", e);
78            }
79            throw e;
80          }
81          finally {
82            close(jsIn, log);
83          }
84        }
85      }
86      catch( Exception e ) {
87        if( log.isErrorEnabled() ) {
88          log.error("There was an error with the system id defined in '"+manifestSystemId+"'"+(reader != null ? " line number '"+reader.getLineNumber()+"'" : "")+".");
89        }
90        throw e;
91      }
92      finally {
93        close(manifestIn, log);
94      }
95    }
