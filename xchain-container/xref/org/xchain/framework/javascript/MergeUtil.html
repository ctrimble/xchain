37  public class MergeUtil {
38    private static Logger log = LoggerFactory.getLogger(MergeUtil.class);
39    
40    public static URL[] parseManifest(String manifestSystemId) throws IOException {
41      List<URL> jsUrlList = new ArrayList<URL>();
42      LineNumberReader manifestReader = null;
43      InputStream manifestIn = null;
44      try {
45        // get the url for the absolute system id.
46        URL manifestUrl = UrlFactory.getInstance().newUrl(manifestSystemId);
47        // create a reader that will allow us to track the line number of the system id we are working with.
48        manifestReader = new LineNumberReader(new InputStreamReader(manifestIn = manifestUrl.openStream()));
49        String jsSystemId;
50        while( ( jsSystemId = manifestReader.readLine() ) != null ) {
51          String absoluteJsSystemId = URI.create(manifestSystemId).resolve(jsSystemId).toString();
52          jsUrlList.add(UrlFactory.getInstance().newUrl(absoluteJsSystemId));
53        }
54      } finally {
55        close(manifestReader, log);
56        close(manifestIn, log);
57      }
58      
59      return jsUrlList.toArray(new URL[jsUrlList.size()]);
60    }
