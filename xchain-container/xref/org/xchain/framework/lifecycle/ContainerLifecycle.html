16  package org.xchain.framework.lifecycle;
17  
18  import java.net.URL;
19  import java.util.Iterator;
20  
21  import javax.servlet.ServletContext;
22  
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  import org.xchain.framework.net.protocol.resource.ResourceUrlConnection;
26  import org.xchain.framework.net.protocol.resource.ServletContextUrlTranslationStrategy;
27  import org.xchain.framework.net.strategy.BaseUrlUrlTranslationStrategy;
28  import org.xchain.framework.net.strategy.CompositeUrlTranslationStrategy;
29  
30  /**
31   * @author John Trimble
32   * @author Devon Tackett
33   * @author Josh Kennedy
34   */
35  @LifecycleClass(uri="http://www.xchain.org/framework/lifecycle")
36  public class ContainerLifecycle {
37    public static final Logger log = LoggerFactory.getLogger( ContainerLifecycle.class );
38    public static final String SERVLET_CONTEXT_ATHORITY = "servlet-context";
39  
40    private static ServletContext servletContext = null;
41    
42    public static void setServletContext( ServletContext context ) {
43      servletContext = context;
44    }
45  
46    public static ServletContext getServletContext() {
47      return servletContext;
48    }
49  
50    /**
51     * Setup Servlet Context resource access.
52     * @param context
53     */
54    @StartStep(localName="servlet", after="config")
55    public static void startLifecycle( LifecycleContext context ) 
56    {
57      ConfigContext configContext = context.getConfigContext();
58  
59      // configure the servlet-context authority in the URLFactory
60      if( ContainerLifecycle.getServletContext() != null ) {
61  
62        ServletContextUrlTranslationStrategy.setServletContext( ContainerLifecycle.getServletContext() );
63        CompositeUrlTranslationStrategy servletContextStrategy = new CompositeUrlTranslationStrategy();
64  
65        // configure the URLFactory for file monitoring if requested
66        if( configContext.isMonitored() ) {
67  
68          if( log.isDebugEnabled() ) {
69            log.debug( "ServletStep: Monitoring is enabled, configuring URL translation strategies..." );
70          }
71  
72          // configure the resource protocol 'servlet-context' authority for monitoring if available
73          if( !configContext.getWebappUrlList().isEmpty() ) {
74            for( Iterator<URL> it = configContext.getWebappUrlList().iterator(); it.hasNext(); ) {
75              URL baseUrl = it.next();
76              BaseUrlUrlTranslationStrategy baseUrlStrategy =
77                new BaseUrlUrlTranslationStrategy( baseUrl, BaseUrlUrlTranslationStrategy.URL_FACTORY_URL_SOURCE );
78              servletContextStrategy.getTranslatorList().add( baseUrlStrategy );
79              if( log.isDebugEnabled() ) {
80                log.debug( "    Adding webapp URL: " + baseUrl );
81              }
82            }
83          }
84        }
85  
86        // now add the servlet context strategy
87        servletContextStrategy.getTranslatorList().add( new ServletContextUrlTranslationStrategy() );
88  
89        ResourceUrlConnection.registerUrlTranslationStrategy( SERVLET_CONTEXT_ATHORITY,
90                                                              servletContextStrategy );
91      }
92    }
93    
94    @StopStep(localName="servlet")
95    public static void stopLifecycle( LifecycleContext context ) 
96    {
97      // cleanup the servlet-context authority in the URLFactory
98      if( ContainerLifecycle.getServletContext() != null ) {
99        ResourceUrlConnection.getUrlTranslationStrategyMap().keySet().remove( SERVLET_CONTEXT_ATHORITY );
100       ServletContextUrlTranslationStrategy.setServletContext( null );
101       ContainerLifecycle.setServletContext( null );
102     }
103   }
104 }
