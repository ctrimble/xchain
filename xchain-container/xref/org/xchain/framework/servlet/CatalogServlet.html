40  /**
41   * This servlet can be used to bind servlet requests to xchains inside of an application.  To do this, this servlet tasks the following action:
42   * <ol>
43   *   <li>The requested url is translated into the name of a catalog.  This catalog is then loaded.</li>
44   *   <li>The requested method is translated into a qname.  This command is loaded from the catalog.</li>
45   *   <li>The servlet request and response are used to create a JXPathContext.</li>
46   *   <li>The JXPathcContext is used to execute the command.</li>
47   * </ol>
48   *
49   * @author Mike Moulton
50   * @author Devon Tackett
51   * @author Christian Trimble
52   * @author John Trimble
53   * @author Josh Kennedy
54   */
55  public class CatalogServlet
56    extends HttpServlet
57  {
58    public static Logger log = LoggerFactory.getLogger(CatalogServlet.class);
59  
60    public static final String DEFAULT_BASE_CATALOG_NAME = "resource://" + ContainerLifecycle.SERVLET_CONTEXT_ATHORITY + "/";
61    public static final String BASE_CATALOG_NAME_PARAM = "base-catalog-name";
62  
63    protected String baseCatalogName = null;
64  
65    public void init( ServletConfig config )
66      throws ServletException
67    {
68      super.init(config);
69  
70      // read the base catalog name.
71      baseCatalogName = config.getInitParameter(BASE_CATALOG_NAME_PARAM);
72      if( baseCatalogName == null ) { baseCatalogName = DEFAULT_BASE_CATALOG_NAME; }
73  
74      // log the config.
75      if( log.isDebugEnabled() ) {
76        log.debug("Base Catalog Name: "+baseCatalogName);
77      }
78    }
79    
80    protected void service(HttpServletRequest request, HttpServletResponse response)
81    	throws ServletException, java.io.IOException
82    {
83      ServletThreadContext context = new ServletThreadContext(request, response);
84      try {
85        ThreadLifecycle.getInstance().startThread(context);
86      }
87      catch( LifecycleException le ) {
88        throw new ServletException("Could not start the servlet thread due to an exception.");
89      }
90      try {
91        // All methods may be valid commands for a catalog.  No need to break out the usage into different
92        // method calls.
93        executeCommand( request, response );
94      }
95      finally {
96        try {
97          ThreadLifecycle.getInstance().stopThread(context);
98        }
99        catch( LifecycleException le ) {
100         if( log.isWarnEnabled() ) {
101           log.warn("An exception was thrown while cleaning up a thread.", le);
102         }
103       }
104     }
105 
106   }

108   /**
109    * Execute the proper XChain based on the incoming {@link HttpServletRequest} and sending output on the {@link HttpServletResponse}.
110    * The path on the request determines which catalog to use while the method of the request determines which command to execute.
111    * 
112    * @param request The incoming request.
113    * @param response The outgoing response.
114    */
115   public void executeCommand( HttpServletRequest request, HttpServletResponse response )
116     throws ServletException, IOException
117   {
118 
119     try {
120       // get the catalog name for the request.
121       String catalogName = catalogNameForRequest( request);
122 
123       // get the command name for the request.
124       QName commandName = commandNameForRequest( request );
125 
126       if( log.isDebugEnabled() ) {
127         log.debug("Executing '"+commandName+"' in catalog '"+catalogName+"'.");
128       }
129       
130       // create the context for the request.
131       JXPathContext context = jXPathContext( request, response );
132 
133       // execute the command.
134       boolean result = CatalogFactory.getInstance().getCatalog(catalogName).getCommand(commandName).execute(context);
135 //      boolean result = CommandUtil.execute(catalogName, commandName, context);
136       
137       if( response.isCommitted() == false && result == false ) {
138         // it looks like we didn't do anything...
139       }
140     }
141     catch( CatalogNotFoundException cae ) {
142       if( log.isDebugEnabled() ) {
143         log.debug("Could not find xchain catalog.", cae);
144       }
145       response.sendError( HttpServletResponse.SC_NOT_FOUND );
146     }
147     catch( CommandNotFoundException coe ) {
148       if( log.isDebugEnabled() ) {
149         log.debug("Could not find xchain command.", coe);
150       }
151       response.sendError( HttpServletResponse.SC_NOT_FOUND );
152     }
153     catch( Exception e ) {
154       
155       if (log.isErrorEnabled()) {
156         log.error("Unable to execute xchain.", e);
157       }
158       
159       throw new ServletException(e);
160     }
161   }

163   /**
164    * Get the catalog based on the requested path.
165    * 
166    * @param request The incoming HttpServletRequest.
167    * 
168    * @return The full requested catalog name.
169    */
170   protected String catalogNameForRequest( HttpServletRequest request)
171   {
172     StringBuffer sb = new StringBuffer();
173 
174     // append the base catalog name.
175     sb.append( baseCatalogName );
176 
177     String servletPath = request.getServletPath();
178 
179     if( servletPath.startsWith("/") && baseCatalogName.endsWith("/") ) {
180       sb.append(servletPath.replaceAll("\\A/(.*)\\Z", "$1"));
181     }
182     else if( !servletPath.startsWith("/") && !baseCatalogName.endsWith("/") ) {
183       sb.append("/").append(servletPath);
184     }
185     else {
186       sb.append(servletPath);
187     }
188 
189     // return the catalog name.
190     return sb.toString();
191   }

193   /**
194    * Get the command requested based on the request method.
195    * 
196    * @param request The incoming HttpServletRequest.
197    * 
198    * @return The QName of the requested command.
199    */
200   protected QName commandNameForRequest( HttpServletRequest request )
201   {
202     return new QName( Constants.URI, request.getMethod().toLowerCase() );
203   }

205   /**
206    * Build a new JXPathContext and populate it with the given request and response.
207    * 
208    * @param request The incoming HttpServletRequest.
209    * @param response The outgoing HttpServletResponse.
210    * 
211    * @return A JXPath which contains the servlet request and response.
212 
213    */
214   protected JXPathContext jXPathContext( HttpServletRequest request, HttpServletResponse response )
215   {
216     JXPathContext context = JXPathContext.newContext( new HashMap() );
217 
218     ((QNameVariables)context.getVariables()).declareVariable( new QName( Constants.URI, Constants.CONTEXT ), getServletConfig().getServletContext() );
219     ((QNameVariables)context.getVariables()).declareVariable( new QName( Constants.URI, Constants.REQUEST ), request );
220     ((QNameVariables)context.getVariables()).declareVariable( new QName( Constants.URI, Constants.RESPONSE ), response );
221 
222     return context;
223   }
