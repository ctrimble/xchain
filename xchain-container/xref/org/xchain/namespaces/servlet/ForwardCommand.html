16  package org.xchain.namespaces.servlet;
17  
18  import javax.servlet.ServletContext;
19  import javax.servlet.ServletRequest;
20  import javax.servlet.ServletResponse;
21  import javax.servlet.RequestDispatcher;
22  
23  import org.apache.commons.jxpath.JXPathContext;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  import org.xchain.Command;
27  import org.xchain.annotations.Attribute;
28  import org.xchain.annotations.AttributeType;
29  import org.xchain.annotations.Element;
30  import org.xchain.annotations.PrefixMapping;
31  
32  /**
33   * @author Mike Moulton
34   * @author Jason Rose
35   * @author Josh Kennedy
36   */
37  @Element(localName="forward")
38  public abstract class ForwardCommand
39    implements Command
40  {
41  
42    public static Logger log = LoggerFactory.getLogger(ForwardCommand.class);
43    
44    @Attribute(
45        localName="path",
46        type=AttributeType.JXPATH_VALUE
47       )
48    public abstract String getPath(JXPathContext context);
49  
50    @Attribute(
51        localName="servlet-context",
52        type=AttributeType.JXPATH_VALUE,
53        defaultValue="$" + Constants.DEFAULT_PREFIX + ":" + Constants.CONTEXT,
54        defaultPrefixMappings={@PrefixMapping(uri=Constants.URI, prefix=Constants.DEFAULT_PREFIX)}
55       )
56    public abstract ServletContext getServletContext(JXPathContext context);
57  
58    @Attribute(
59        localName="response",
60        type=AttributeType.JXPATH_VALUE,
61        defaultValue="$" + Constants.DEFAULT_PREFIX + ":" + Constants.RESPONSE,
62        defaultPrefixMappings={@PrefixMapping(uri=Constants.URI, prefix=Constants.DEFAULT_PREFIX)}
63       )
64    public abstract ServletResponse getResponse(JXPathContext context);
65  
66    @Attribute(
67        localName="request",
68        type=AttributeType.JXPATH_VALUE,
69        defaultValue="$" + Constants.DEFAULT_PREFIX + ":" + Constants.REQUEST,
70        defaultPrefixMappings={@PrefixMapping(uri=Constants.URI, prefix=Constants.DEFAULT_PREFIX)}
71       )
72    public abstract ServletRequest getRequest(JXPathContext context);
73  
74    public boolean execute( JXPathContext context )
75      throws Exception
76    {
77      ServletContext servletContext = getServletContext(context);
78      RequestDispatcher dispatcher = servletContext.getRequestDispatcher(getPath(context));
79      dispatcher.forward(getRequest(context), getResponse(context));
80  
81      return false;
82    }
83  }
