17  
18  import static java.lang.annotation.ElementType.METHOD;
19  import static java.lang.annotation.RetentionPolicy.RUNTIME;
20  
21  import java.lang.annotation.Documented;
22  import java.lang.annotation.Retention;
23  import java.lang.annotation.Target;
24  import java.lang.annotation.Inherited;
25  import javax.xml.XMLConstants;

27  /**
28   * <p>Binds an abstract method to an xml attribute for a Command.  When a command has an abstract
29   * method marked with the annotation, XChains will engineer all of the code needed to access the value described
30   * by the attribute from a JXPathContext.</p>
31   *
32   * <p>Methods marked with this annotation must meet the following requirements:</p>
33   * <ul>
34   *   <li>The method must be abstract.</li>
35   *   <li>The method must take a JXPathContext as its first parameter.</li>
36   *   <li>If the method has two parameters, the second parameter must be of type java.lang.Class and the annotation's type attribute must be AttributeType.JXPATH_VALUE.</li>
37   *   <li>The method must not have more than two parameters.</li>
38   * </ul>
39   *
40   * <h3>Basic usage:</h3>
41   * <p>To bind a method to an attribute, two things must be specified.  First the name of the attribute must be specified.  Secondly, the way to treat the attributes value must be specified.
42   * This can be accomplished with the localName and type attributes of this annotation.  Here is an example:</ p>
43   * <pre>
44   *   &#064;Attribute(localName="attr", type=AttributeType.JXPATH_VALUE)
45   *   public abstract Object getAttr(JXPathContext context);
46   * </pre>
47   * <p>This method binds the attribute "attr" in the default namespace and defines its value to be an XPath.  The engineered method will include a call to
48   * JXPathContext.getValue(String, Class) where the xpath is the value of the attribute "attr" and the class is java.lang.Object.</p>
49   *
50   * <h3>Defining a default value for an attribute:</h3>
51   * <p>When defining a default value for an attribute, the defaultValue attribute of the annotation must be defined.  If the default value uses any prefixes, then the
52   * defaultPrefixMappings attribute must also be included on the annotation and it must include the prefixes required for the default value.  For example:</p>
53   * <pre>
54   *   &#064;Attribute(localName="attr", type=AttributeType.JXPATH_VALUE, defaultValue="example:function()", defaultPrefixMappings={"xmlns:example='http://www.xchain.org/example'"}
55   *   public abstract Object getAttr(JXPathContext context);
56   * </pre>
57   * <p>If the attribute "attr" is not specified, then the default prefix mappings will be registered with the context and the attribute will be evaluated as though it had
58   * the value "example:function()".  Once the evaluation is done, then all of the default prefix mappings will be unregistered.</p>
59   *
60   * <h3>Marking an attribute as required:</h3>
61   * <p>When defining that an attribute is required, the required attribute of the annotation must be defined to be true.  If this part of the annotation is defined, then an XChain will
62   * not load if the attribute is missing.  For example:</p>
63   * <pre>
64   *   &#064;Attribute(localName="attr", type=AttributeType.JXPATH_VALUE, required=true}
65   *   public abstract Object getAttr(JXPathContext context);
66   * </pre>
67   * <p>If the attribute "attr" is not specified, then any attempt to load a catalog containing that command will fail with an exception.</p>
68   * 
69   * @author Mike Moulton
70   * @author Christian Trimble
71   */
72  @Target(METHOD)
73  @Retention(RUNTIME)
74  @Inherited
75  @Documented
76  public @interface Attribute
77  {
78    /**
79     * <p>Defines the local name component of this attribute's QName.</p>
80     *
81     * @return the local name component of this attribute's QName.</p>
82     */
83    String localName();
84  
85    /**
86     * <p>Defines the namespace uri component of this attributes QName.</p>
87     *
88     * @return the namespace uri component of this attribute's QName. 
89     */
90    String namespaceUri() default XMLConstants.NULL_NS_URI;
91  
92    /**
93     * <p>Defines the type of this attributes value.</p>
94     *
95     * @return the type of this attributes value.
96     */
97    AttributeType type() default AttributeType.JXPATH_VALUE;
98  
99    /**
100    * <p>Defines the default value for this attribute.</p>
101    *
102    * @return the default value for this attribute.
103    */
104   String defaultValue() default "";
105 
106   /**
107    * <p>Defines the prefix mapping context for the default value.  The values of this array are defined like namespace prefixes in
108    * an xml document.  Here is an example that binds the prefixes "one" and "two" to the namespaces "http://www.xchain.org/one" and "http://www.xchain.org/two", respectively.</p>
109    * <pre>
110    *   defaultPrefixMapping={"xmlns:one='http://www.xchain.org/one'", "xmlns:two='http://www.xchain.org/two'"}
111    * </pre>
112    *
113    * @return the prefix mapping context for the default value.
114    */
115   PrefixMapping[] defaultPrefixMappings() default {};
116 
117   /**
118    * <p>Defines if this attribute is required.  A value of true means that this attribute is required.  A value of false means that this attribute is not required.  The default
119    * value if false.</p>
120    *
121    * @return true if this attribute is required, false otherwise.
122    */
123   boolean required() default false;
124 }
