56    private class RuleSerializationHandler
57      extends HandlerWrapper
58    {
59      private int depth = 0;
60  
61      public void startElement( String namespaceUri, String localName, String qName, Attributes atts)
62        throws SAXException
63      {
64        depth++;
65  
66        // if we are not including the containing element, then do not pass it on.  Otherwise, pass this event on to super.
67        if( depth == 1 && includeContainingElement || depth > 1 ) {
68          super.startElement( namespaceUri, localName, qName, atts );
69        }
70  
71        // if the depth is ever less than or equal to zero, then things are broken.
72        else if( depth <= 0 ) {
73          throw new IllegalStateException("Unmatched start and end elements detected.");
74        }
75      }
76  
77      public void endElement( String namespaceUri, String localName, String qName )
78        throws SAXException
79      {
80        // if we are not including the containing element, then do not pass it on.
81        if( depth == 1 && includeContainingElement || depth > 1 ) {
82          super.endElement( namespaceUri, localName, qName );
83        }
84  
85        // decrement the depth.
86        depth--;
87  
88        // close down the document and return control to the rule.
89        if( depth <= 0 ) {
90          Digester digester = getDigester();
91  
92          // reset the handlers.
93          digester.setCustomContentHandler(oldCustomContentHandler);
94          oldCustomContentHandler = null;
95          if( digester instanceof ExtendedDigester ) {
96            ((ExtendedDigester)digester).setCustomLexicalHandler(oldCustomLexicalHandler);
97            oldCustomLexicalHandler = null;
98          }
99  
100         // close all of the namespace mappings.
101         Iterator currentNamespaceIterator = digester.getCurrentNamespaces().entrySet().iterator();
102         while( currentNamespaceIterator.hasNext() ) {
103           Map.Entry currentNamespace = (Map.Entry)currentNamespaceIterator.next();
104           handler.endPrefixMapping((String)currentNamespace.getKey());
105         }
106 
107         // terminate the document.
108         super.endDocument();
109 
110         // pass control back to the digester.
111         digester.endElement( namespaceUri, localName, qName );
112       }
113     }
114   }

117   protected RuleSerializationHandler handler = null;

135    * after this rule will be sent.

137   protected boolean includeContainingElement = false;

139   public void begin(String namespaceUri, String name, Attributes attributes)
140     throws Exception
141   {
142     // get the digester.
143     Digester digester = getDigester();
144 
145     // store the old content handlers.
146     oldCustomContentHandler = digester.getCustomContentHandler();
147     if( digester instanceof ExtendedDigester ) {
148       oldCustomLexicalHandler = ((ExtendedDigester)digester).getCustomLexicalHandler();
149     }
150 
151     // create the handler.
152     handler = new RuleSerializationHandler();
153  
154     // set up the handlers that will do the serialization.
155     handler.setWrappedHandler(newHandler());
156 
157     // set the new handlers.
158     digester.setCustomContentHandler(handler);
159     if( digester instanceof ExtendedDigester ) {
160       ((ExtendedDigester)digester).setCustomLexicalHandler(handler);
161     }
162 
163     // push the buffer onto the stack.
164     digester.push(buffer);
165 
166     // start the document.
167     handler.startDocument();
168 
169     // set all of the namespaces that are defined on the digester.
170     Iterator currentNamespaceIterator = digester.getCurrentNamespaces().entrySet().iterator();
171     while( currentNamespaceIterator.hasNext() ) {
172       Map.Entry currentNamespace = (Map.Entry)currentNamespaceIterator.next();
173       handler.startPrefixMapping((String)currentNamespace.getKey(), (String)currentNamespace.getValue());
174     }
175 
176     // send the current element to the handler.
177     if( digester.getNamespaceAware() ) {
178       handler.startElement(namespaceUri, name, digester.getCurrentElementName(), attributes);
179     }
180     else {
181       handler.startElement(namespaceUri, name, name, attributes);
182     }
183   }

191   /**
192    * Returns a new handler for the body of a serializer block.
193    */
194   protected ContentHandler newHandler()
195     throws Exception
196   {
197     Serializer serializer = newSerializer();
198     if( method.toLowerCase().equals("html") ) {
199       // wrap the html handler.
200       XHtmlHandler xhtmlHandler = new XHtmlHandler();
201       xhtmlHandler.setWrappedHandler(serializer);
202       return xhtmlHandler;
203     }
204     else {
205       return serializer.asContentHandler();
206     }
207   }

209   /**
210    * Sets up the wrapped handler.
211    */
212   protected Serializer newSerializer()
213   {
214     Properties outputProperties = OutputPropertiesFactory.getDefaultMethodProperties( method );
215 
216     if ( method.toLowerCase().equals("html") ) {
217       if (indent == null) indent = true; // default to indenting mode
218       outputProperties.setProperty( "media-type", "text/html" );
219       outputProperties.setProperty( "doctype-system", "http://www.w3.org/TR/html4/loose.dtd" );
220       outputProperties.setProperty( "doctype-public", "-//W3C//DTD HTML 4.01 Transitional//EN" );
221     }
222     else if ( method.toLowerCase().equals("xhtml") ) {
223       if (indent == null) indent = true; // default to indenting mode
224       outputProperties.setProperty( "media-type", "application/xhtml+xml" );
225       outputProperties.setProperty( "omit-xml-declaration", "yes" ); // todo: should be browser sensitive
226       outputProperties.setProperty( "doctype-system", "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd" );
227       outputProperties.setProperty( "doctype-public", "-//W3C//DTD XHTML 1.0 Transitional//EN" );
228     }
229     else if ( method.toLowerCase().equals("xml") ) {
230       outputProperties.setProperty( "media-type", "text/xml" );
231     }
232     else if ( method.toLowerCase().equals("text") ) {
233       outputProperties.setProperty( "media-type", "text/plain" );
234     }
235 
236     if ( Boolean.TRUE.equals( indent )) {
237       outputProperties.setProperty( "indent", "yes" );
238       outputProperties.setProperty( "{http://xml.apache.org/xalan}indent-amount", "2" );
239     }
240     else {
241       outputProperties.setProperty( "indent", "no" );
242     }
243 
244     try {
245       Serializer serializer = SerializerFactory.getSerializer( outputProperties );
246       serializer.setWriter(newStringWriter());
247       return serializer;
248     } catch (org.apache.xml.serializer.utils.WrappedRuntimeException e) {
249       log.error("Serializer threw wrapped exception", e.getException());
250       throw e;
251     }
