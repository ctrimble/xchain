40  public class TemplateCompiler
41  {
42    private static Logger log = LoggerFactory.getLogger(TemplateCompiler.class);
43  
44    public static final String SOURCE_VERSION = "1.5";
45    public static final String SOURCE_ENCODING = "UTF-8";
46    public static final String TARGET_VERSION = "1.5";
47  
48    /** The class loader for the catalog. */
49    private TemplateClassLoader templateClassLoader;
50  
51    /** The compiler that we will use for creating classes. */
52    private JavaCompiler compiler;
53  
54    /** The resource reader where source files will be stored. */
55    private WrappedResourceReader resourceReader;
56  
57    /** The settings for the compiler. */
58    private JavaCompilerSettings settings;
59  
60    public void init(ClassLoader classLoader)
61    {
62      this.templateClassLoader = new TemplateClassLoader(classLoader);
63  
64      resourceReader = new WrappedResourceReader(new MemoryResourceReader(), templateClassLoader);
65  
66      settings = new JavaCompilerSettings();
67      settings.setSourceVersion(SOURCE_VERSION);
68      settings.setSourceEncoding(SOURCE_ENCODING);
69      settings.setTargetVersion(TARGET_VERSION);
70  
71      compiler = new JavaCompilerFactory().createCompiler("eclipse");
72    }
73  
74    /**
75     * Compiles the source result.
76     *
77     * @throws SAXException if there is a compilation error in the source file.
78     */
79    public Class compileTemplate( SourceResult result )
80      throws SAXException
81    {
82      // add the resource to the reader.
83      try {
84        resourceReader.add(result.getSourceResourceName(), result.getSource().getBytes(SOURCE_ENCODING));
85      }
86      catch( UnsupportedEncodingException uee ) {
87        throw new SAXException("Could not build source in the encoding '"+SOURCE_ENCODING+"'.", uee);
88      }
89  
90      // create the resource store.
91      ResourceStore resourceStore = new MemoryResourceStore();
92  
93      // compile the result.
94      CompilationResult compilationResult = compiler.compile(new String[] {result.getSourceResourceName()}, resourceReader, resourceStore, templateClassLoader, settings);
95  
96      // handle any errors.
97      if( compilationResult.getErrors().length > 0 ) {
98        if( log.isDebugEnabled() ) {
99          log.debug("Source that would not complile:\n"+result.getSource());
100         for( CompilationProblem error : compilationResult.getErrors() ) {
101           log.debug("Error ("+error.getStartLine()+", "+error.getStartColumn()+"):"+error.getMessage());
102         }
103       }
104       // TODO: Make this message better.
105       StringBuilder stringBuilder = new StringBuilder();
106       stringBuilder.append("Could not compile template source. Errors:\n");
107       for( CompilationProblem error : compilationResult.getErrors() ) {
108         stringBuilder.append("("+error.getStartLine()+", "+error.getStartColumn()+"):"+error.getMessage()+"\n");
109       }
110       throw new SAXException(stringBuilder.toString());
111     }
112 
113     // handle any errors.
114     if( log.isDebugEnabled() ) {
115       for( CompilationProblem warning : compilationResult.getWarnings() ) {
116         log.debug("Compilation warning("+warning.getStartLine()+", "+warning.getStartColumn()+"):"+warning.getMessage());
117       }
118     }
119 
120     byte[] compiledBytes = resourceStore.read(result.getClassResourceName());
121 
122     templateClassLoader.publicDefineClass(result.getClassName(), compiledBytes, 0, compiledBytes.length);
123     Class templateClass = null;
124 
125     try {
126       templateClass = templateClassLoader.loadClass(result.getClassName());
127     }
128     catch( Exception e ) {
129       throw new SAXException("Could not load the class '"+result.getClassName()+"'.", e);
130     }
131     
132     return templateClass;
133   }

143     public void publicDefineClass( String name, byte[] b, int off, int len )

150   public static class WrappedResourceReader
151     implements ResourceReader
152   {
153     protected MemoryResourceReader wrapped;
154     protected ClassLoader classLoader;
155     public WrappedResourceReader( MemoryResourceReader wrapped, ClassLoader classLoader ) { this.wrapped = wrapped; this.classLoader = classLoader; }
156     public void add( String name, byte[] data ) {
157       wrapped.add(name, data);
158     }
159     public byte[] getBytes(String name) {
160       byte[] result = wrapped.getBytes(name);
161       if( result == null ) {
162         try {
163         int bufferSize = 0;
164         byte[] buffer = new byte[10000];
165         int read = 0;
166 
167         InputStream source = classLoader.getResourceAsStream(name);
168         while( (read = source.read(buffer, bufferSize, buffer.length)) != -1 ) {
169           bufferSize += read;
170           if( bufferSize >= buffer.length ) {
171             break;
172           }
173         }
174 
175         result = new byte[bufferSize];
176         for( int i = 0; i < bufferSize; i++ ) {
177           result[i] = buffer[i];
178         }
179 
180         }
181         catch( Exception e ) {
182           e.printStackTrace();
183           return null;
184         }
185       }
186       return result;
187     }
188 
189     public boolean isAvailable( String name)
190     {
191       boolean result = wrapped.isAvailable(name);
192       if( !result ) {
193         result = (classLoader.getResource(name) != null);
194       }
195       return result;
196     }
197   }
