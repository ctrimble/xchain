28  /**
29   * <p>This class represents a lifecycle step that is defined using annotations.</p>
30   *
31   * <p>The start method for this lifecycle step can be defined using setStartMethod(Method).  The start method should take at most one argument of type LifecycleContext.  If the
32   * method throws an exception of a type other than LifecycleException, then the exception will be wrapped with a LifecycleException.</p>
33   *
34   * <p>The end method for this lifecycle step can be defined using setStopMethod(Method).  The stop method should take at most one argument of type LifecycleContext.  If the
35   * stop method throws an exception of a type other than RuntimeException, then the exception will be wrapped with a RuntimeException.</p>
36   *
37   * <p>If either the start or stop method is an instance method, then a static method for accessing the lifecycle class must be defined.  This method will be called before
38   * calling an instance start or stop method, to get access to the lifecycle instance.</p>
39   *
40   * @author Christian Trimble
41   * @author John Trimble
42   */

113   /**
114    * <p>If the start method is defined, then this method will invoke the start method, otherwise this method does nothing.</p>
115    * 
116    * @param context The current LifecycleContext.
117    * 
118    * @throws LifecycleException If an exception is encountered starting this Lifecycle step.
119    */
120   public void startLifecycle(LifecycleContext context, ConfigDocumentContext configDocContext)
121     throws LifecycleException
122   {
123     if( startMethod != null ) {
124       Map<String, String> savedPrefixMappings = null;
125       try {
126         // map prefixes
127         if( this.prefixMappings != null ) {
128           savedPrefixMappings = new HashMap<String, String>();
129           definePrefixMappings(configDocContext, prefixMappings, savedPrefixMappings);
130         }
131         invokeStepMethod(startMethod, lifecycleAccessor, context, configDocContext);
132       }
133       catch( InvocationTargetException ite ) {
134         if( ite.getCause() instanceof LifecycleException ) {
135           throw (LifecycleException)ite.getCause();
136         }
137         else if( ite.getCause() instanceof RuntimeException ) {
138           throw (RuntimeException)ite.getCause();
139         }
140         else if( ite.getCause() instanceof Error ) {
141           throw (Error)ite.getCause();
142         }
143         else {
144           throw new LifecycleException("The start method "+qName+" threw an unknown exception type.", ite);
145         }
146       }
147       catch( Exception e ) {
148         throw new LifecycleException("Could not execute lifecycle start step method "+qName+".", e);
149       } finally {
150         // unmap prefixes 
151         if( this.prefixMappings != null ) {
152           undefinePrefixMappings(configDocContext, prefixMappings, savedPrefixMappings);
153         }
154       }
155     }
156   }

158   private static void definePrefixMappings(ScopedJXPathContextImpl context, Map<String,String> prefixMappings, Map<String,String> savedPrefixMappings) 
159   {
160     for( Map.Entry<String, String> entry : prefixMappings.entrySet() ) {
161       String uri = context.getNamespaceURI(entry.getKey());
162       // save old mapping
163       if( uri != null )
164         savedPrefixMappings.put(entry.getKey(), uri);
165       // set prefix mapping
166       context.registerNamespace(entry.getKey(), entry.getValue());
167     }
168   }
169   
170   private static void undefinePrefixMappings(ScopedJXPathContextImpl context, Map<String,String> prefixMappings, Map<String,String> savedPrefixMappings) {
171     // unregister prefix mappings
172     for( Map.Entry<String, String> entry : prefixMappings.entrySet() ) {
173       context.registerNamespace(entry.getKey(), null);
174     }
175     // restore saved mappings
176     for( Map.Entry<String, String> entry : savedPrefixMappings.entrySet() ) {
177       context.registerNamespace(entry.getKey(), entry.getValue());
178     }
179   }

181   /**
182    * <p>If the end method is defined, then this method will invoke the start method, otherwise this method does nothing.</p>
183    * 
184    * @param context The current LifecycleContext.
185    */
186   public void stopLifecycle(LifecycleContext context)
187   {
188     try {
189       if( stopMethod != null ) {
190         invokeStepMethod(stopMethod, lifecycleAccessor, context, null);
191       }
192     }
193     catch( InvocationTargetException ite ) {
194       if( ite.getCause() instanceof RuntimeException ) {
195         throw (RuntimeException)ite.getCause();
196       }
197       else if( ite.getCause() instanceof Error ) {
198         throw (Error)ite.getCause();
199       }
200       else {
201         throw new RuntimeException("The stop method "+qName+" threw an unknown exception type.", ite);
202       }
203     }
204     catch( RuntimeException re ) {
205       throw re;
206     }
207     catch( Exception e ) {
208       throw new RuntimeException("The lifecycle stop step "+qName+" threw an exception.", e);
209     }
210   }

212   private static Object invokeStepMethod(Method stepMethod, Method lifecycleAccessor, LifecycleContext lifecycleContext, ConfigDocumentContext configContext) 
213     throws IllegalArgumentException, IllegalAccessException, InvocationTargetException, LifecycleException 
214   {
215     Object lifecycleObject = null;
216     if( !Modifier.isStatic(stepMethod.getModifiers()) ) {
217       if( lifecycleAccessor == null ) {
218         throw new LifecycleException("Cannot execute step method '"+stepMethod.getName()+"' of class '"+stepMethod.getDeclaringClass().getName()+"' without an accessor method.");
219       }
220       lifecycleObject = lifecycleAccessor.invoke(null);
221 
222       if( lifecycleObject == null ) {
223         throw new LifecycleException("The accessor method '"+lifecycleAccessor+"' of class '"+lifecycleAccessor.getDeclaringClass().getName()+"' returned a null lifecycle object.");
224       }
225     }
226     
227     Class<?>[] paramTypes = stepMethod.getParameterTypes();
228     Object[] values = new Object[paramTypes.length];
229     for( int i = 0; i < paramTypes.length; i++ ) {
230       if( paramTypes[i].isInstance(lifecycleContext) )
231         values[i] = lifecycleContext;
232       else if( paramTypes[i].isInstance(configContext) ) 
233         values[i] = configContext;
234     }
235     return stepMethod.invoke(lifecycleObject, values);
236   }
