84     * Start an execution stack on the current thread.
85     * 
86     * @param globalContext The global JXPathContext for this execution.
87     */
88    public static void startExecution(JXPathContext globalContext)
89    {
90      // Wrap the incoming context in a global context
91      executionContextTl.set(new ScopedJXPathContextImpl(globalContext, globalContext.getContextBean(), Scope.execution));
92      executionContextStack.push(new GlobalExecutionContext());
93    }

95    /**
96     * End all execution for this thread.  All contexts will be cleared.
97     * 
98     * @throws ExecutionException If an exception was thrown during execution it will be wrapped in an ExecutionException.
99     */
100   public static void endExecution()
101     throws ExecutionException
102   {
103     ExecutionException executionException = null;
104     // Check if an exception was encountered during execution.
105     if( executionContextStack.peek().exceptionContext != null ) {
106       // An exception was found.  Create a new ExecutionException with a trace to the source of the exception.
107       executionException = new ExecutionException("An exception was thrown during the execution.", executionContextStack.peek().exceptionContext.trace, executionContextStack.peek().exceptionContext.exception);
108     }
109     // Clear all stacks.
110     executionContextStack.clear();
111     suspendedExecutionContextStack.clear();
112     executionTraceStack.clear();
113     suspendedExecutionTraceStack.clear();
114     chainContextStack.clear();
115     suspendedChainContextStack.clear();
116     
117     // End the execution context.
118     endContext(executionContextTl.get());
119     executionContextTl.set(null);
120     if( executionException != null ) {
121       // Throw the ExecutionException if one was created.
122       throw executionException;
123     }
124   }

126   /**
127    * Returns a string representation of the current execution.
128    */
129   private static String stateString()
130   {
131     StringBuilder builder = new StringBuilder();
132     builder.append("Execution Context Stacks:").append(executionContextStack.size()).append("/").append(suspendedExecutionContextStack.size());
133     builder.append(" Execution Trace Stacks:").append(executionTraceStack.size()).append("/").append(suspendedExecutionTraceStack.size());
134     builder.append(" Local Context Stacks:").append(chainContextStack.size()).append("/").append(suspendedChainContextStack.size());
135     builder.append(" Global Context:").append(executionContextTl.get()!=null);
136     return builder.toString();
137   }

142     builder.append("Execution Context Stack:\n");
143     for( ExecutionContext executionContext : executionContextStack.toList() ) {
144       builder.append("  ").append(executionContext.toString()).append("\n");
145     }
146     builder.append("Suspended Execution Context Stack:\n");

149     }
150     return builder.toString();

161   /**
162    * <p>Signals that the execute method of a command has been called.  If this command is registered with a catalog, then
163    * a new local context is created based on the current context that was passed in.</p>
164    *
165    * <p>NOTE: This command is used by the engineering framework and should not be called directly by
166    * command implementations.</p>
167    *
168    * @param command the command that is being executed.
169    * @param context the current local context or the global context.
170    * @return the current local context.  If the command is registered with a catalog, then the context returned is
171    * the new local context for this command.  Otherwise, the context returned will be the same as the context passed in.
172    */
173   public static JXPathContext startCommandExecute( Command command, JXPathContext context )
174   {
175     JXPathContext localContext = null;
176 
177     // push this command onto the stack.
178     executionContextStack.push(new CommandExecutionContext(command));
179 
180     if( isLocalContextBoundary(command) ) {
181       // create the new local context.
182       localContext = new ScopedJXPathContextImpl( executionContextTl.get(), context.getContextBean(), context.getContextPointer(), Scope.chain );
183 
184       // push the local context.
185       chainContextStack.push(localContext);
186 
187       startExecutionTrace(command);
188     }
189     else if( !(context instanceof ScopedJXPathContextImpl ) ) {
190       throw new IllegalStateException("Initial call to command that is not registered with a catalog.");
191     }
192 
193     updateExecutionTraceLocation();
194     
195     // get the local context.
196     context = getCurrentContext();
197     if (command instanceof EngineeredCommand) {
198       // Define prefix mappings for engineered commands.
199       definePrefixMappings(context, (EngineeredCommand)command);
200     }    
201 
202     // return the context.
203     return context;
204   }

206   /**
207    * Returns true if the command represents a local context boundary, false otherwise.
208    *
209    * @return true if the command represents a local context boundary, false otherwise.
210    */
211   private static boolean isLocalContextBoundary( Command command )
212   {
213     if( !(command instanceof Registerable ) ) {
214       return false;
215     }
216 
217     return ((Registerable)command).isRegistered();
218   }

220   /**
221    * <p>Signals that we are creating a local context for context bean
222    * that has the same variable and function scope as the last context
223    * bean.</p>
224    * 
225    * <p>The user can safely call this method.</p>
226    * 
227    * @param context the current local context.
228    * @param contextPointer A Pointer to the new context bean.
229    * 
230    * @return the new local context that shares variables and namespace context with the previous local context,
231    * but it has a new context bean.
232    */
233   public static JXPathContext startContextPointer( JXPathContext context, Pointer contextPointer )
234   {
235     // make sure that this is the current local context.
236     testCurrentLocalContext(context);
237 
238     // create the relative context.
239     JXPathContext localContext = context.getRelativeContext( contextPointer );
240 
241     // push the context onto the stack.
242     chainContextStack.push(localContext);
243 
244     // return the new local context.
245     return localContext;
246   }

248   /**
249    * <p>Executed at the end of a command's execute method if it is not a filter or at the end of a
250    * command's postProcess method if it is a filter.  This method will change the current local
251    * context if the current command is registered with a catalog.</p>
252    *
253    * <p>NOTE: This command is used by the engineering framework and should not be called directly by
254    * command implementations.</p>
255    *
256    * @param context the current local context.
257    * @return the previous local context on the context stack.
258    */
259   public static JXPathContext endCommandExecute(Command command, JXPathContext context)
260   {
261     if (command instanceof EngineeredCommand) {
262       // Undefine prefix mappings for engineered commands.
263       undefinePrefixMappings(context, (EngineeredCommand)command);
264     }
265     
266     // make sure that we are managing the stacks correctly.
267     testCurrentLocalContext(context);
268     testCurrentCommand(command);
269 
270     boolean isFilter = command instanceof Filter;
271 
272     // if we are in a filter, the suspend everything.
273     if( isFilter ) {
274       suspendedExecutionContextStack.push(executionContextStack.pop());
275 
276       // if this is a local context boundary, then suspend the context.
277       if( isLocalContextBoundary( command ) ) {
278         suspendedChainContextStack.push(chainContextStack.pop());
279         suspendExecutionTrace();
280       }
281     }
282     // otherwise, this is a just a command, so pop it's state off of the stacks.
283     else {
284       // remove the command from the stack.
285       executionContextStack.pop();
286 
287       // if this was a context boundary, then update the execution trace and the local context stack.
288       if( isLocalContextBoundary( command ) ) {
289         endContext(chainContextStack.pop());
290         stopExecutionTrace();
291       }
292     }
293 
294     // update the current location of the execution trace.
295     updateExecutionTraceLocation();
296 
297     // return the current context.
298     return getCurrentContext();
299   }

315    * <p>Marks the end of a commands post process method.</p>
316    * <p>NOTE: This command is used by the engineering framework and should not be called directly by
317    * command implementations.</p>
318    *
319    * @param context the current local context.
320    * @return the previous local context on the context stack.
321    */
322   public static JXPathContext endCommandPostProcess(Command command, JXPathContext context)
323   {    
324     if (command instanceof EngineeredCommand) {
325       // Undefine prefix mappings for engineered commands.
326       undefinePrefixMappings(context, (EngineeredCommand)command);
327     }    
328     
329     // make sure that we are managing the stacks correctly.
330     testCurrentLocalContext(context);
331     testCurrentCommand(command);
332 
333     // move the command to the top of the suspended command stack.
334     executionContextStack.pop();
335 
336     // otherwise, this is a just a command, so pop it's state off of the stacks.
337     if( isLocalContextBoundary( command ) ) {
338       endContext(chainContextStack.pop());
339       stopExecutionTrace();
340     }
341 
342     // update the current location of the execution trace.
343     updateExecutionTraceLocation();
344 
345     // return the current context.
346     return getCurrentContext();
347   }

362   /**
363    * Resumes the top item on the suspended context stack.
364    *
365    * <p>NOTE: This command is used by the engineering framework and should not be called directly by
366    * command implementations.</p>
367    *
368    * @param context the current local context.
369    * @return the context that was resumed.
370    */
371   public static JXPathContext startCommandPostProcess(Command command, JXPathContext context)
372   {
373     try {
374     // make sure that we are managing the stacks correctly.
375     testCurrentLocalContext(context);
376 
377     // move the top of the suspended command stack to the top of the command stack.
378     executionContextStack.push(suspendedExecutionContextStack.pop());
379     testCurrentCommand(command);
380 
381     // move the top of the 
382     if( isLocalContextBoundary( command ) ) {
383       chainContextStack.push(suspendedChainContextStack.pop());
384       resumeExecutionTrace();
385     }
386 
387     // update the current location of the execution trace.
388     updateExecutionTraceLocation();
389     }
390     catch( Exception e ) {
391       e.printStackTrace();
392       throw new RuntimeException(e);
393     }
394     
395     // Get the current context.
396     context = getCurrentContext();
397     if (command instanceof EngineeredCommand) {
398       // Define prefix mappings for engineered commands.
399       definePrefixMappings(context, (EngineeredCommand)command);
400     }
401 
402     // return the current context.
403     return context;
404   }

406   /**
407    * Add all custom prefixes for the given command to the context.  PrefixMappings are stored in a stack.  The last prefix mapping that
408    * is defined must be the first one undefined.  Any prefixes that conflict with the command's prefixes will have their original
409    * values saved and will be restored once the prefix mapping is undefined for the given command.
410    * 
411    * @param context The context being used.
412    * @param command The command whose custom prefixes are to be used.
413    */
414   public static void definePrefixMappings( JXPathContext context, EngineeredCommand command )
415   {
416     PrefixMappingContext prefixContext = null;
417     // get the first prefix mapping.
418     if( prefixMappingStack.get().size() > 0 && prefixMappingStack.get().getFirst().isContextFor( context, command ) ) {
419       prefixContext = prefixMappingStack.get().getFirst();
420       prefixContext.setCallCount(prefixContext.getCallCount()+1);
421     }
422     else {
423       // create a new prefix context.
424       prefixContext = new PrefixMappingContext(context, command);
425 
426       // place the prefix on the top of the prefix context stack.
427       prefixMappingStack.get().addFirst(prefixContext);
428 
429       // iterate over the mappings defined in the command, storing the old values.
430       for( Map.Entry<String, String> entry : command.getPrefixMap().entrySet() ) {
431         prefixContext.getOriginalPrefixMap().put(entry.getKey(), context.getNamespaceURI(entry.getKey()));
432       }
433 
434       // set the new values into the context.
435       for( Map.Entry<String, String>entry : command.getPrefixMap().entrySet() ) {
436         context.registerNamespace( entry.getKey(), entry.getValue() );
437       }
438     }
439   }

441   /**
442    * Remove all custom prefixes for the given command from the context.  Prefix mappings are stored in a stack.  The
443    * latest prefix mapping that has been defined must be the first one undefined.  When a prefix mapping is undefined
444    * any prefixes that conflicted with the command's prefixes will have their original values restored.
445    * 
446    * @param context The context being used.
447    * @param command The command whose custom prefixes are to be undefined.
448    */
449   public static void undefinePrefixMappings( JXPathContext context,  EngineeredCommand command )
450   {
451     // get the first item from the stack.
452     PrefixMappingContext prefixContext = prefixMappingStack.get().getFirst();
453 
454     if( !prefixContext.isContextFor( context, command ) ) {
455       throw new RuntimeException("CommandUtil.undefinePrefixMapping called on wrong context and command.");
456     }
457 
458     // if this is a duplicate mapping call, then just decrement the call count.
459     if( prefixContext.getCallCount() > 0 ) {
460       prefixContext.setCallCount(prefixContext.getCallCount() - 1);
461     }
462 
463     // otherwise, we need to remove the mappings and remove the context.
464     else {
465 
466       // set the original values into the context.
467       for( Map.Entry<String, String> entry : prefixContext.getOriginalPrefixMap().entrySet() ) {
468         context.registerNamespace( entry.getKey(), entry.getValue() );
469       }
470 
471       // remove the prefix mapping from the context stack.
472       prefixMappingStack.get().removeFirst();
473 
474       // if the stack is empty, then remove the stack for this thread.
475       if( prefixMappingStack.get().size() == 0 ) {
476         prefixMappingStack.remove();
477       }
478     }
479   }

494   /**
495    * <p>Pushes an execution trace onto the top of the execution trace stack.</p>
496    *
497    * @param command the command to start an execution trace for.
498    */
499   private static void startExecutionTrace( Command command )
500   {
501     String systemId = null;
502     QName qName = null;
503 
504     if( command instanceof Registerable ) {
505       Registerable registerable = (Registerable)command;
506       if( registerable.isRegistered() ) {
507         systemId = registerable.getSystemId();
508         qName = registerable.getQName();
509       }
510     }
511 
512     // add a new locator for this systemId and qName to the top of the stack.
513     executionTraceStack.push(new ExecutionTraceElement(systemId, qName, null));
514   }

541   /**
542    * Updates the current location of the top entry of the execution trace stack.
543    */
544   private static void updateExecutionTraceLocation()
545   {
546     if( !executionTraceStack.isEmpty() ) {
547     Command command = ((CommandExecutionContext)executionContextStack.peek()).command;
548     Locator locator = null;
549 
550     if( command instanceof Locatable ) {
551       locator = ((Locatable)command).getLocator();
552     }
553     else {
554       LocatorImpl locatorImpl = new LocatorImpl();
555       locatorImpl.setLineNumber(0);
556       locatorImpl.setColumnNumber(0);
557       locatorImpl.setSystemId("UNKNOWN_LOCATION");
558       locator = locatorImpl;
559     }
560 
561     executionTraceStack.peek().setLocator(locator);
562     }
563   }

578   /**
579    * Called by engineered commands to notify that an exception is propagating from an execute(JXPathContext) method.
580    */
581   public static void exceptionThrown( Command command, Exception exception )
582   {
583     // make sure that we are in the current command.
584     testCurrentCommand(command);
585 
586     // get the parent context.
587     ExecutionContext parentExecutionContext = executionContextStack.peek(1);
588     ExecutionContext executionContext = executionContextStack.peek();
589     ExceptionContext exceptionContext = executionContext.exceptionContext;
590 
591     // if the exception thrown is the same exception on the current node, then move it to the parent.
592     if( exceptionContext != null && exceptionContext.exception == exception ) {
593       parentExecutionContext.exceptionContext = exceptionContext;
594     }
595 
596     // if the exception is different, but one of the filters claimed to handle it, then we need to create a new context with no cause.
597     else if( exceptionContext == null || exceptionContext.handled ) {
598       parentExecutionContext.exceptionContext = new ExceptionContext(exception, getExecutionTrace(), null);
599     }
600 
601     // if the exception thrown is different and it was not handled, then it must be a cause.
602     else {
603       parentExecutionContext.exceptionContext = new ExceptionContext(exception, getExecutionTrace(), exceptionContext);
604     }
605   }

607   public static void exceptionHandled( Command command, Exception exception )
608   {
609     // make sure that we are in the current command.
610     testCurrentCommand(command);
611 
612     // mark the exception as handled.
613     executionContextStack.peek(1).exceptionContext.handled = true;
614   }

643   /**
644    * Returns a copy of the current execution trace stack.  This list contains all of the currently executing
645    * xchains starting with the current chain and anding with the first chain that was called.
646    */
647   public static List<ExecutionTraceElement> getExecutionTrace()
648   {
649     // we need clone the entries in this list, since the locations are changing.
650     // we will reuse the list, since toList() creates a new list.
651     List<ExecutionTraceElement> executionTrace = executionTraceStack.toList();
652     for( int i = 0; i < executionTrace.size(); i++ ) {
653       executionTrace.set(i, new ExecutionTraceElement(executionTrace.get(i)));
654     }
655     return executionTrace;
656   }

658   /**
659    * Tests that the context passed in is the current local context.
660    */
661   private static void testCurrentLocalContext( JXPathContext context )
662   {
663     if( (chainContextStack.isEmpty() && executionContextTl.get() != context) ) {
664       throw new IllegalStateException("The global context should have been passed to Execution.startCommandPostProcess().");
665     }
666     else if (!chainContextStack.isEmpty() && chainContextStack.peek() != context ) {
667       throw new IllegalStateException("The local context passed to Execution.XXXLocalContext() was not the current local context.");
668     }
669   }

675   {
676     if( ((CommandExecutionContext)executionContextStack.peek()).command != command ) {
677       throw new IllegalStateException("The command passed to Execution.XXXCommand() was not the current command.");
