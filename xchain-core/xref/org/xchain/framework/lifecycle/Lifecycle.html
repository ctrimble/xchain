69  /**
70   * The main class of the xchain framework.  When the lifecycle is started the LifecycleStepScanner will scan for LifecycleSteps
71   * in the current context class loader.  All LifecycleSteps will then be started with the Lifecycle.  When the Lifecycle stops
72   * all the LifecycleSteps started when the Lifecycle started will be stopped in a LIFO manner.

79   * @see LifecycleStep

102   /**
103    * Start the lifecycle.  Loading all lifecycle steps found by the LifecycleStepScanner.
104    *
105    * @see LifecycleStepScanner
106    */
107   public static void startLifecycle()
108     throws LifecycleException
109   {
110     try {
111       ThreadLifecycle.getInstance().getCCLPolicy().bindCCL();
112       
113     synchronized( Lifecycle.class ) {
114       // if xchains is currently running, then throw a lifecycle exception.
115       if( isRunning() ) {
116         throw new LifecycleException("Start Lifecycle called while xchains is running.");
117       }
118 
119       try {
120         // create a new Lifecycle context.
121         context = new LifecycleContext();
122         context.setClassLoader(new LifecycleClassLoader(Thread.currentThread().getContextClassLoader()));
123 
124         // scan for the lifecycle listeners.
125         startLifecycleSteps();
126         // Clear the scanner cache now that all the lifecycles are done using it.
127         ScannerLifecycle.getInstance().clearCache();
128       }
129       catch( LifecycleException le ) {
130         context = null;
131         throw le;
132       }
133       catch( Throwable t ) {
134         context = null;
135         throw new LifecycleException("Could not start the lifecycle due to a throwable.", t);
136       }
137     }
138 
139     }
140     finally {
141       ThreadLifecycle.getInstance().getCCLPolicy().unbindCCL();
142     }
143   }

150   {
151     synchronized( Lifecycle.class ) {

191   /**
192    * Start all lifecycle steps found by the LifecycleStepScanner.
193    * 
194    * @throws LifecycleException If an exception was encountered attempting to start the lifecycle steps.
195    * @see LifecycleStepScanner
196    */
197   private static void startLifecycleSteps()
198     throws LifecycleException
199   {
200     // Create a scanner to find the lifecycle steps in the current classpath.
201     LifecycleStepScanner scanner = new LifecycleStepScanner(context);
202     try {
203       scanner.scan();
204       lifecycleStepList = scanner.getLifecycleStepList();
205     }
206     catch( ScanException se ) {
207       throw new LifecycleException("An exception was thrown while scanning for lifecycle steps.", se);
208     }
209 
210     if( log.isInfoEnabled() ) {
211       StringBuilder message = new StringBuilder();
212       message.append("Found ").append(lifecycleStepList.size()).append(" lifecycle steps:\n");
213       for( LifecycleStep lifecycleStep : lifecycleStepList ) {
214         message.append("  ").append(lifecycleStep.getQName()).append("\n");
215       }
216       log.info(message.toString());
217     }
218 
219     // Start all the found lifecycle steps.
220     ListIterator<LifecycleStep> iterator = lifecycleStepList.listIterator();
221     try {
222       while( iterator.hasNext() ) {
223         LifecycleStep lifecycleStep = iterator.next();
224 
225         if( log.isInfoEnabled() ) {
226           log.info("Starting Lifecycle Step '"+lifecycleStep.getQName()+"'.");
227         }
228         lifecycleStep.startLifecycle(context, Lifecycle.configDocumentContext);
229         if( log.isInfoEnabled() ) {
230           log.info("Finished Lifecycle Step '"+lifecycleStep.getQName()+"'.");
231         }
232       }
233     }
234     catch( LifecycleException le ) {
235       if( log.isErrorEnabled() ) {
236         log.error("Stopping the lifecycle startup due to a lifecycle exception.", le);
237       }
238       iterator.previous();
239       while( iterator.hasPrevious() ) {
240         LifecycleStep lifecycleStep = iterator.previous();
241         try {
242           lifecycleStep.stopLifecycle(context);
243         }
244         catch( Throwable t ) {
245           if( log.isWarnEnabled() ) {
246             log.warn("An exception was thrown while stopping a lifecycle exception.", t);
247           }
248         }
249       }
250 
251       // clear the lifecycle step list.
252       lifecycleStepList.clear();
253 
254       // we should throw the lifecycle exception here.
255       throw le;
256     } 
257     finally {
258       // make sure the configuration DOM gets garbage collected, no reason to keep it around once everyone is configured.
259       Lifecycle.configDocumentContext = null;
260     }
261     
262   }

264   /**
265    * Stop all current lifecycle steps.  The first step to be stopped is the last one started.
266    */
267   private static void stopLifecycleSteps()
268   {
269     ListIterator<LifecycleStep> iterator = lifecycleStepList.listIterator(lifecycleStepList.size());
270     while( iterator.hasPrevious() ) {
271       LifecycleStep step = iterator.previous();
272       try {
273         if( log.isInfoEnabled() ) {
274           log.info("Stopping Lifecycle Step '"+step.getQName()+"'.");
275         }
276         step.stopLifecycle(context);
277         if( log.isInfoEnabled() ) {
278           log.info("Finished Lifecycle Step '"+step.getQName()+"'.");
279         }
280       }
281       catch( Throwable t ) {
282         if( log.isWarnEnabled() ) {
283           log.warn("An exception was thrown while stopping a lifecycle exception.", t);
284         }
285       }
286     }
287     lifecycleStepList.clear();
288   }

304   /**
305    * Lifecycle Step to load the xchain configuration.
306    */
307   @StartStep(localName="config", xmlns={"xmlns:config='http://xchain.org/config/1.0'"})
308   public static void startConfiguration(LifecycleContext context, ConfigDocumentContext configDocContext) 
309     throws MalformedURLException 
310   {
311     // Read DOM and set values on configContext
312     ConfigContext configContext = Lifecycle.getLifecycleContext().getConfigContext();
313     Boolean monitor = (Boolean)configDocContext.getValue("/config:config/config:monitor", Boolean.class);
314     if( monitor != null ) configContext.setMonitored(monitor);
315     Integer catalogCacheSize = (Integer)configDocContext.getValue("/config:config/config:catalog-cache-size", Integer.class);
316     if( catalogCacheSize != null ) configContext.setCatalogCacheSize(catalogCacheSize);
317     Integer templateCacheSize = (Integer)configDocContext.getValue("/config:config/config:templates-cache-size", Integer.class);
318     if( templateCacheSize != null ) configContext.setTemplatesCacheSize(templateCacheSize);
319     
320     addUrls(configDocContext, "/config:config/config:resource-base-url/@config:system-id", configContext.getResourceUrlList());
321     addUrls(configDocContext, "/config:config/config:source-base-url/@config:system-id", configContext.getSourceUrlList());
322     addUrls(configDocContext, "/config:config/config:webapp-base-url/@config:system-id", configContext.getWebappUrlList());
323     
324     // configure the URLFactory for file monitoring if requested
325     if( configContext.isMonitored() ) {
326 
327       if( log.isDebugEnabled() ) {
328         log.debug( "Config: Monitoring is enabled, configuring URL translation strategies..." );
329       }
330 
331       // configure the resource protocol 'context-class-loader' authority for monitoring
332       if( !configContext.getResourceUrlList().isEmpty() ) {
333 
334         CompositeUrlTranslationStrategy contextClassLoaderStrategy = new CompositeUrlTranslationStrategy();
335   
336         for( Iterator<URL> it = configContext.getResourceUrlList().iterator(); it.hasNext(); ) {
337           URL baseUrl = it.next();
338           BaseUrlUrlTranslationStrategy baseUrlStrategy =
339             new BaseUrlUrlTranslationStrategy( baseUrl, BaseUrlUrlTranslationStrategy.URL_FACTORY_URL_SOURCE );
340           contextClassLoaderStrategy.getTranslatorList().add( baseUrlStrategy );
341           if( log.isDebugEnabled() ) {
342             log.debug( "    Adding resource URL: " + baseUrl );
343           }
344         }
345   
346         // now add the standard context class loader strategy
347         contextClassLoaderStrategy.getTranslatorList().add( new ContextClassLoaderUrlTranslationStrategy() );
348   
349         // override the standard strategy with the new composite strategy
350         ResourceUrlConnection.registerUrlTranslationStrategy( ResourceUrlConnection.CONTEXT_CLASS_LOADER_ATHORITY,
351                                                               contextClassLoaderStrategy );
352       }
353     }
354   }

395    * Sets the default SAX, XSLT, and DOM implementations to use on the XmlFactoryLifecycle for those that are not 
396    * already set. To override any of these defaults, create a chain that runs before this one and sets the factories as
397    * desired on the XmlFactoryLifecycle.

399    * @param context
400    */
401   @StartStep(localName="default-xml-factory", before={"xml-factory-lifecycle"})
402   public static void startDefaultXmlFactory(LifecycleContext context)
403   {
404     // add the default xml parser factory step.
405     if( getSaxParserFactoryFactory(DEFAULT_SAX_PARSER_FACTORY_NAME) == null ) {
406       saxParserFactoryCreated = true;
407       putSaxParserFactoryFactory(DEFAULT_SAX_PARSER_FACTORY_NAME, new DefaultSaxParserFactoryFactory());
408     }
409 
410     // add the default xml transformer factory.
411     if( getTransformerFactoryFactory(DEFAULT_TRANSFORMER_FACTORY_NAME) == null ) {
412       transformerFactoryCreated = true;
413       putTransformerFactoryFactory(DEFAULT_TRANSFORMER_FACTORY_NAME, new DefaultTransformerFactoryFactory());
414     }
415 
416     // add the default document builder factory
417     if( getDocumentBuilderFactory(DEFAULT_DOCUMENT_BUILDER_FACTORY_NAME) == null ) {
418       documentBuilderFactoryCreated = true;
419       putDocumentBuilderFactoryFactory(DEFAULT_DOCUMENT_BUILDER_FACTORY_NAME, new DefaultDocumentBuilderFactoryFactory());
420     }
421 
422     // add the default xml transformer factory.
423     if( getTransformerFactoryFactory(XALAN_FACTORY_NAME) == null && factoryClassExists(XALAN_FACTORY_NAME, XALAN_FACTORY_CLASS_NAME) ) {
424       xalanFactoryCreated = true;
425       putTransformerFactoryFactory(XALAN_FACTORY_NAME, new BasicTransformerFactoryFactory(XALAN_FACTORY_CLASS_NAME));
426     }
427 
428     // add the default xml transformer factory.
429     if( getTransformerFactoryFactory(XSLTC_FACTORY_NAME) == null && factoryClassExists(XSLTC_FACTORY_NAME, XSLTC_FACTORY_CLASS_NAME) ) {
430       xsltcFactoryCreated = true;
431       putTransformerFactoryFactory(XSLTC_FACTORY_NAME, new BasicTransformerFactoryFactory(XSLTC_FACTORY_CLASS_NAME));
432     }
433 
434     // add the default xml transformer factory.
435     if( getTransformerFactoryFactory(SAXON_FACTORY_NAME) == null && factoryClassExists(SAXON_FACTORY_NAME, SAXON_FACTORY_CLASS_NAME) ) {
436       saxonFactoryCreated = true;
437       putTransformerFactoryFactory(SAXON_FACTORY_NAME, new BasicTransformerFactoryFactory(SAXON_FACTORY_CLASS_NAME));
438     }
439 
440     if( getTransformerFactoryFactory(JOOST_FACTORY_NAME) == null && factoryClassExists(JOOST_FACTORY_NAME, JOOST_FACTORY_CLASS_NAME) ) {
441       joostFactoryCreated = true;
442       putTransformerFactoryFactory(JOOST_FACTORY_NAME, new BasicTransformerFactoryFactory(JOOST_FACTORY_CLASS_NAME));
443     }

522   /**
523    * Creates a new ConfigContext of the DOM produced by parsing the indicated resource.
524    * 
525    * @throws IOException 
526    * @throws SAXException 
527    * @throws ParserConfigurationException 
528    */
529   private static ConfigDocumentContext createConfigurationContext(String resource) 
530     throws LifecycleException 
531   {
532     try {
533       Object contextBean = null;
534       URL configUrl = Thread.currentThread().getContextClassLoader().getResource(resource);
535       if( configUrl == null ) {
536         // we can't find the config so we just an instance of Object for the ConfigDocumentContext.
537         if( log.isDebugEnabled() )
538           log.debug("Cannot find configuration at resource '"+resource+"'.");
539         contextBean = new Object();
540       } else {
541         // we have a config, create a DOM out of it and use it for the ConfigDocumentContext.
542         if( log.isDebugEnabled() ) {
543           log.debug("Loading xchain config file for url: "+configUrl.toExternalForm());
544         }
545         
546         // get the document builder.
547         DocumentBuilder documentBuilder = XmlFactoryLifecycle.newDocumentBuilder();
548         
549         InputSource configInputSource = UrlSourceUtil.createSaxInputSource(configUrl);
550         Document document = documentBuilder.parse(configInputSource);
551         contextBean = document;
552       }
553       // create the context
554       ConfigDocumentContext configDocumentContext = new ConfigDocumentContext(null, contextBean, Scope.chain);
555       configDocumentContext.setConfigUrl(configUrl);
556       configDocumentContext.setLenient(true);
557       return configDocumentContext;
558     } catch( Exception e ) {
559       throw new LifecycleException("Error loading configuration from resource '"+resource+"'.", e);
560     }
561   }

563   private static void addUrls(ConfigDocumentContext configDocContext, String urlxpath, List<URL> urlList) 
564     throws MalformedURLException 
565   {
566     Iterator<?> urlStringIterator = configDocContext.iterate(urlxpath);
567     while( urlStringIterator.hasNext() ) {
568       String urlString = urlStringIterator.next().toString();
569       if( !"".equals(urlString) ) {
570         urlList.add(new URL(urlString));
571       }
572     }

575   private static boolean factoryClassExists( QName factoryName, String className )
576   {
577     boolean result = false;
578     try {
579       Thread.currentThread().getContextClassLoader().loadClass(className);
580       result = true;
581     }
582     catch( ClassNotFoundException cnfe ) {
583       if( log.isInfoEnabled() ) {
584         log.info("Not creating default transformer factory for '"+factoryName+"' because its driver class '"+className+"' is not in the context class loader.");
585       }
586     }
587     return result;
