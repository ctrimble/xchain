50   * This scanner scans the context class loader for lifecycle steps and resolves their dependencies.  After the scan() method has been invoked, a call to
51   * getLifecycleStepList() will return all of the lifecycles steps in their proper order.

53   * @author Christian Trimble
54   * @author Devon Tackett

56   * @author Josh Kennedy

59    extends AbstractScanner

63    protected static final Pattern NAMESPACE_DEFINITION_MAPPING_PATTERN = Pattern.compile("xmlns[:]([\\w-.]+)=['\"]([^'\"]+)['\"]");
64    protected static final String VALID_STEP_METHOD_SIGNATURES = "Lifecycle step methods must have one of the following parameter signatures: (), (LifecycleContext), (ConfigDocumentContext), (LifecycleContext, ConfigDocumentContext).";

79    /**
80     * Scans the context class loader for lifecycle steps and resolves their dependencies.  After this method is called, a call to getLifecycleStepList
81     * will return all of the lifecycle steps in their proper order.
82     *
83     * @throws ScanException this exception is thrown if there is a cyclic dependency among the lifecycle steps or if there is a problem creating an instance
84     * of one of the lifecycle steps.
85     */
86    public void scan()
87      throws ScanException
88    {
89      try {
90        dependencySorter = new DependencySorter<QName>(new LexicographicQNameComparator());
91        lifecycleStepList = new ArrayList<LifecycleStep>();
92        lifecycleStepMap = new HashMap<QName, LifecycleStep>();
93        classPool = EngineeringUtil.createClassPool(classLoader);
94        lifecycleStepCtClass = classPool.get("org.xchain.framework.lifecycle.LifecycleStep");
95  
96        super.scan();
97  
98        try {
99          List<QName> lifecycleStepQNameList = dependencySorter.sort();
100         for( QName lifecycleStepQName : lifecycleStepQNameList ) {
101           LifecycleStep lifecycleStep = lifecycleStepMap.get(lifecycleStepQName);
102           if( lifecycleStep != null ) {
103             lifecycleStepList.add(lifecycleStep);
104             //try {
105               //lifecycleStepList.add(lifecycleStepClass.newInstance());
106             //}
107             //catch( Exception e ) {
108               //throw new ScanException("Could not create instance of lifecycle step '"+lifecycleStepClass.getName()+"'.", e);
109             //}
110           }
111         }
112       }
113       catch( DependencyCycleException dce ) {
114         throw new ScanException("There is at least one cycle in the lifecycle step dependencies.", dce);
115       }
116     }
117     catch( NotFoundException nfe ) {
118       throw new ScanException("The definition of a required class could not be found.", nfe);
119     }
120     finally {
121       dependencySorter = null;
122       classPool = null;
123       lifecycleStepCtClass = null;
124     }
125   }

137   public void scanNode( ScanNode node )
138     throws ScanException
139   {
140     try {
141       if( isLoadableClassFile(node.getResourceName()) ) {
142         String className = toClassName(node);
143 
144         CtClass scannedCtClass = null;
145         boolean requiresAnnotationScanning = false;
146 
147         try {
148           scannedCtClass = classPool.get(className);
149           requiresAnnotationScanning = hasAnnotation(scannedCtClass, LifecycleClass.class);
150         }
151         catch( Exception e ) {
152           if( log.isDebugEnabled() ) {
153             log.debug("Could not scan file '"+node.getResourceName()+"' due to an exception.", e);
154           }
155           return;
156         }
157         if( requiresAnnotationScanning ) {
158           // get the actual class.
159           Class lifecycleClass = context.getClassLoader().loadClass(className);
160 
161           // get the lifecycle annotation.
162           LifecycleClass lifecycleAnnotation = (LifecycleClass)lifecycleClass.getAnnotation(LifecycleClass.class);
163           
164           // get the uri for the annotations found in the class.
165           String namespaceUri = lifecycleAnnotation.uri();
166 
167           // scan the lifecycle class for a lifecycle accessor method.
168           Method accessorMethod = null;
169           int accessorMethodCount = 0;
170           for( Method method : lifecycleClass.getDeclaredMethods() ) {
171             if( Modifier.isStatic(method.getModifiers()) && hasAnnotation(method, LifecycleAccessor.class) ) {
172               accessorMethod = method;
173               accessorMethodCount++;
174             }
175           }
176 
177           if( accessorMethodCount > 1 ) {
178             throw new ScanException("The class '"+lifecycleClass+"' has "+accessorMethodCount+" static methods that are annotated with the LifecycleAccessor annotation."+
179                      " Lifecycle classes should have at most one accessor method.");
180           }
181 
182           // scan the class for for start and stop steps.
183           for( Method method : lifecycleClass.getMethods() ) {
184             boolean isStartStep = hasAnnotation(method, StartStep.class);
185             boolean isStopStep = hasAnnotation(method, StopStep.class);
186             boolean isStatic = Modifier.isStatic(method.getModifiers());
187             boolean isConfigStep = isDocumentConfigStep(method);
188 
189             if( isStartStep && isStopStep ) {
190               throw new ScanException("The class '"+lifecycleClass+"' has a lifecycle method '"+method.getName()+"' that has both a StartStep and a StopStep annotations.");
191             }
192 
193             // if this is not a start or stop step, then move on.
194             if( !isStartStep && !isStopStep ) {
195               continue;
196             }
197             
198             // ASSERT: The method has a single step annotation.
199 
200             assertProperStepMethodSignature(lifecycleClass, method);
201 
202             // ASSERT: The method has the proper signature.
203 
204             String localName = null;
205             QName qName = null;
206             Set<QName> beforeSet = null;
207             Set<QName> afterSet = null;
208 
209             if( isStartStep ) {
210               // get the start step annotation.
211               StartStep startStep = method.getAnnotation(StartStep.class);
212               localName = startStep.localName();
213               qName = new QName(namespaceUri, localName);
214               beforeSet = toQNameSet(startStep.before(), namespaceUri);
215               afterSet = toQNameSet(startStep.after(), namespaceUri);
216             }
217             else if( isStopStep ) {
218               StopStep stopStep = method.getAnnotation(StopStep.class);
219               localName = stopStep.localName();
220               qName = new QName(namespaceUri, localName);
221               afterSet = toQNameSet(stopStep.before(), namespaceUri);
222               beforeSet = toQNameSet(stopStep.after(), namespaceUri);
223             }
224             
225             // add implicit dependency for steps that need configuration information--this will insure those steps
226             // run only after the step that creates the configuration information they need.
227             if( isStartStep && isConfigStep ) 
228               afterSet.add(QName.valueOf("{http://www.xchain.org/framework/lifecycle}create-config-document-context"));
229 
230             // ASSERT: the qName, beforeSet, and afterSet have all been set from the perspective of a start step.
231             
232             Map<String, String> prefixMappings = null;
233             if( isStartStep ) {
234               // get the start step annotation.
235               StartStep startStep = method.getAnnotation(StartStep.class);
236               prefixMappings = buildPrefixMappings(startStep.xmlns(), new HashMap<String, String>());
237             }
238             
239             // ASSERT: prefix mappings set for the start step.
240 
241             LifecycleStep step = lifecycleStepMap.get(qName);
242             if( step == null ) {
243               step = new AnnotationLifecycleStep(qName);
244             }
245 
246             AnnotationLifecycleStep annotationStep = (AnnotationLifecycleStep)step;
247 
248             // set the accessor method on the annotation step.
249             annotationStep.setLifecycleAccessor(accessorMethod);
250 
251             // make sure that the annotation step does not already have a step defined.
252             if( isStartStep && annotationStep.getStartMethod() != null ) {
253               throw new ScanException("The class '"+lifecycleClass+"' has more than one start step for QName '"+qName+"'.");
254             }
255             else if( isStopStep && annotationStep.getStopMethod() != null ) {
256               throw new ScanException("The class '"+lifecycleClass+"' has more than one stop step for QName '"+qName+"'.");
257             }
258 
259             // ASSERT: There is not a duplicate start or stop step for this qName.
260 
261             if( isStartStep ) {
262               annotationStep.setStartMethod(method);
263               annotationStep.setStartMethodPrefixMappings(prefixMappings);
264             }
265             else if (isStopStep) {
266               annotationStep.setStopMethod(method);
267             }
268 
269             // add the lifecycle step to the map.
270             lifecycleStepMap.put(qName, annotationStep);
271 
272             // add information about the step to the sorter.
273             dependencySorter.add(qName);
274             for( QName before : beforeSet ) {
275               dependencySorter.addDependency(qName, before);
276             }
277             for( QName after : afterSet ) {
278               dependencySorter.addDependency(after, qName);
279             }
280           }
281            
282         }
283       }
284     }
285     catch( Exception e ) {
286       throw new ScanException("Failed to scan file '"+node.getResourceName()+"' due to an exception.", e);
287     }
288   }

290   private static Map<String, String> buildPrefixMappings(String[] prefixDefinitions, Map<String, String> prefixUriMap) throws ScanException {
291     for( String prefixDefinition : prefixDefinitions ) {
292       Matcher matcher = NAMESPACE_DEFINITION_MAPPING_PATTERN.matcher(prefixDefinition); 
293       if( !matcher.matches() ) {
294         throw new ScanException("Invalid prefix definition: "+prefixDefinition);
295       }
296       String prefix = matcher.group(1);
297       String uri = matcher.group(2);
298       prefixUriMap.put(prefix, uri);
299     }
300     return prefixUriMap;
301   }

303   private static void assertProperStepMethodSignature(Class<?> lifecycleClass, Method method) throws ScanException {
304     // check the parameters on the method, to make sure it is a valid method signature.
305     Class<?>[] parameterTypes = method.getParameterTypes();
306     
307     // make sure that the parameter list is valid.  If it isn't, tell the user about the problem.
308     if( parameterTypes.length == 1 && (parameterTypes[0] != LifecycleContext.class && parameterTypes[0] != ConfigDocumentContext.class) ) {
309       throw new ScanException("The class '"+lifecycleClass+"' has a lifecycle method '"+method.getName()+"' that has an illegal signature."+
310                               " "+VALID_STEP_METHOD_SIGNATURES);
311     }
312     if( parameterTypes.length == 2 && (parameterTypes[0] != LifecycleContext.class || parameterTypes[1] != ConfigDocumentContext.class) ) {
313       throw new ScanException("The class '"+lifecycleClass+"' has a lifecycle method '"+method.getName()+"' that has an illegal signature."+
314           " "+VALID_STEP_METHOD_SIGNATURES);
315     }
316     if( parameterTypes.length > 2 ) {
317       throw new ScanException("The class '"+lifecycleClass+"' has a lifecycle method '"+method.getName()+"' that has an illegal signature."+
318           " "+VALID_STEP_METHOD_SIGNATURES);
319     }
320   }

322   private static boolean isDocumentConfigStep(Method stepMethod) {
323     for( Class<?> type : stepMethod.getParameterTypes() ) {
324       if( type.equals(ConfigDocumentContext.class) )
325         return true;
326     }
327     return false;
328   }

330   private static Set<QName> toQNameSet( String[] qNameArray, String defaultUri )
331     throws ScanException
332   {
333     Set<QName> qNameSet = new HashSet<QName>();
334     for( String qNameString : qNameArray ) {
335       QName qName = null;
336       if( qNameString.startsWith("{") ) {
337         qName = QName.valueOf(qNameString);
338       }
339       else if( qNameString.matches("[A-Za-z][-A-Za-z0-9._]*") ) {
340         qName = new QName( defaultUri, qNameString );
341       }
342       else {
343         throw new ScanException("The qname string '"+qNameString+"' does not appear to be a valid qname or local name.");
344       }
345       qNameSet.add(qName);
346     }
347     return qNameSet;
348   }
