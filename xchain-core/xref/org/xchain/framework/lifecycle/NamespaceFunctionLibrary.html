37  /**
38   * <p>This class handles methods for an xchain namespace.  During the scanning process, methods with the org.xchain.annotations.Function annotation are added
39   * to instances of this class.</p>
40   *
41   * <p>The implementation of this class is based on the implementation of ClassFunctions from the jxpath project by Dmitri Plotnikov.</p>
42   *
43   * @author Christian Trimble
44   * @author Josh Kennedy
45   */
46  public class NamespaceFunctionLibrary
47    implements Functions
48  {
49    private static Logger log = LoggerFactory.getLogger(NamespaceFunctionLibrary.class);
50  
51    private static final Object[] EMPTY_ARRAY = new Object[0];
52    private Map<String, Set<MethodInfo>> staticMethodInfoMap = new HashMap<String, Set<MethodInfo>>();
53    private Map<String, Set<MethodInfo>> instanceMethodInfoMap = new HashMap<String, Set<MethodInfo>>();
54    private Map<String, Set<Class>> constructorClassMap = new HashMap<String, Set<Class>>();
55    private String namespace;
56  
57    public NamespaceFunctionLibrary( String namespace )
58    {
59      this.namespace = namespace;
60    }
61  
62    public void addStaticFunction( String localName, Class methodClass, String methodName )
63    {
64      Set<MethodInfo> methodInfoSet = staticMethodInfoMap.get(localName);
65      if( methodInfoSet == null ) {
66        staticMethodInfoMap.put(localName, (methodInfoSet = new LinkedHashSet<MethodInfo>()));
67      }
68      methodInfoSet.add(new MethodInfo(methodClass, methodName));
69    }
70  
71    public void addInstanceFunction( String localName, Class methodClass, String methodName )
72    {
73      Set<MethodInfo> methodInfoSet = instanceMethodInfoMap.get(localName);
74      if( methodInfoSet == null ) {
75        instanceMethodInfoMap.put(localName, (methodInfoSet = new LinkedHashSet<MethodInfo>()));
76      }
77      methodInfoSet.add(new MethodInfo(methodClass, methodName));
78    }
79  
80    public void addSingletonInstanceFunction( String localName, Class methodClass, String methodName, Method accessorMethod )
81    {
82      Set<MethodInfo> methodInfoSet = instanceMethodInfoMap.get(localName);
83      if( methodInfoSet == null ) {
84        instanceMethodInfoMap.put(localName, (methodInfoSet = new LinkedHashSet<MethodInfo>()));
85      }
86      methodInfoSet.add(new MethodInfo(methodClass, methodName, accessorMethod));
87    }
88  
89    public void addConstructorFunction( String localName, Class constructorClass )
90    {
91      Set<Class> constructorClassSet = constructorClassMap.get(localName);
92      if( constructorClassSet == null ) {
93        constructorClassMap.put( localName, (constructorClassSet = new LinkedHashSet<Class>()) );
94      }
95      constructorClassSet.add(constructorClass);
96    }
97  
98    public Set getUsedNamespaces() {
99      return Collections.singleton(namespace);
100   }

102   public Function getFunction( String namespace, String name, Object[] parameters )
103   {
104     // make sure that we are requesting the correct namespace.
105     if (namespace == null && this.namespace != null) {
106       return null;
107     }
108     else if (!namespace.equals(this.namespace)) {
109       return null;
110     }
111 
112     if (parameters == null) {
113       parameters = EMPTY_ARRAY;
114     }
115 
116     Set<Class> constructorClassSet = constructorClassMap.get(name);
117     Constructor constructor;
118     if( constructorClassSet != null ) {
119       for( Class constructorClass : constructorClassSet ) {
120         constructor = MethodLookupUtils.lookupConstructor( constructorClass, parameters );
121         if( constructor != null ) {
122           return new ConstructorFunction(constructor);
123         }
124       }
125     }
126 
127     Set<MethodInfo> methodInfoSet = staticMethodInfoMap.get(name);
128     Method method;
129 
130     if( methodInfoSet != null ) {
131       for( MethodInfo methodInfo : methodInfoSet ) {
132         method = MethodLookupUtils.lookupStaticMethod(methodInfo.getMethodClass(), methodInfo.getMethodName(), parameters);
133         if (method != null) {
134           return new MethodFunction(method);
135         }
136       }
137     }
138 
139     methodInfoSet = instanceMethodInfoMap.get(name);
140 
141     if( methodInfoSet != null ) {
142       for( MethodInfo methodInfo : methodInfoSet ) {
143         Object[] instanceParameters = parameters;
144         Method singletonAccessor = methodInfo.getSingletonAccessor();
145         if( singletonAccessor != null ) {
146           try {
147             Object singleton = singletonAccessor.invoke(null, EMPTY_ARRAY);
148             instanceParameters = new Object[parameters.length+1];
149             instanceParameters[0] = singleton;
150             for( int i = 0; i < parameters.length; i++ ) {
151               instanceParameters[i+1] = parameters[i];
152             }
153           method = MethodLookupUtils.lookupMethod(methodInfo.getMethodClass(), methodInfo.getMethodName(), instanceParameters);
154           if (method != null) {
155             return new SingletonMethodFunction(method, singleton);
156           }
157           }
158           catch( Exception e ) {
159             if( log.isDebugEnabled() ) {
160               log.debug("Could not find singleton for class '"+methodInfo.getMethodClass().getName()+"'.", e);
161             }
162           }
163         }
164         else {
165           method = MethodLookupUtils.lookupMethod(methodInfo.getMethodClass(), methodInfo.getMethodName(), instanceParameters);
166           if (method != null) {
167             return new MethodFunction(method);
168           }
169         }
170       }
171     }
172 
173     return null;
174   }

176   public static class MethodInfo
177   {
178     private Class methodClass;
179     private String methodName;
180     private Method singletonAccessor = null;
181 
182     public MethodInfo(Class methodClass, String methodName)
183     {
184       this.methodClass = methodClass;
185       this.methodName = methodName;
186     }
187 
188     public MethodInfo(Class methodClass, String methodName, Method singletonAccessor )
189     {
190       this( methodClass, methodName );
191       this.singletonAccessor = singletonAccessor;
192     }
193 
194     public int hashCode() {
195       return methodClass.hashCode()/2 + methodClass.hashCode()/2;
196     }
197 
198     public boolean equals( Object o )
199     {
200       if( o instanceof MethodInfo ) {
201         MethodInfo mi = (MethodInfo)o;
202         return methodClass.equals(mi.getMethodClass()) && methodName.equals(mi.getMethodName());
203       }
204       else {
205         return false;
206       }
207     }
208 
209     public Class getMethodClass() { return this.methodClass; }
210     public String getMethodName() { return this.methodName; }
211     public Method getSingletonAccessor() { return this.singletonAccessor; }
212   }

214   public static class SingletonMethodFunction
215     extends MethodFunction
216   {
217     private Object singleton;
218     public SingletonMethodFunction( Method method, Object singleton )
219     {
220       super(method);
221       this.singleton = singleton;
222     }
223     public Object invoke( ExpressionContext context, Object[] parameters )
224     {
225       if( parameters == null ) {
226         parameters = EMPTY_ARRAY;
227       }
228       Object[] newParameters = new Object[parameters.length+1];
229       newParameters[0] = singleton;
230       for( int i = 0; i < parameters.length; i++ ) {
231         newParameters[i+1] = parameters[i];
232       }
233       Object result = super.invoke( context, newParameters );
234 
235       return result;
236     }
237   }
