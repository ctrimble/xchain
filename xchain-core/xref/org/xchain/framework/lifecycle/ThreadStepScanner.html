60    protected static final Pattern NAMESPACE_DEFINITION_MAPPING_PATTERN = Pattern.compile("xmlns[:]([\\w-.]+)=['\"]([^'\"]+)['\"]");
61    protected static final String VALID_STEP_METHOD_SIGNATURES = "Thread step methods must have one of the following parameter signatures: (), (LifecycleContext).";

63    protected List<ThreadStep> threadStepList = null;
64    protected Map<QName, ThreadStep> threadStepMap = null;
65    protected LifecycleContext context = null;

68    protected DependencySorter<QName> dependencySorter = null;

70    public ThreadStepScanner( LifecycleContext context )
71    {
72      super( new MarkerResourceLocator("META-INF/xchain.xml"),  Thread.currentThread().getContextClassLoader() );
73      this.context = context;
74    }

76    /**
77     * Scans the context class loader for lifecycle steps and resolves their dependencies.  After this method is called, a call to getThreadStepList
78     * will return all of the lifecycle steps in their proper order.
79     *
80     * @throws ScanException this exception is thrown if there is a cyclic dependency among the lifecycle steps or if there is a problem creating an instance
81     * of one of the lifecycle steps.
82     */
83    public void scan()
84      throws ScanException
85    {
86      try {
87        dependencySorter = new DependencySorter<QName>(new LexicographicQNameComparator());
88        threadStepList = new ArrayList<ThreadStep>();
89        threadStepMap = new HashMap<QName, ThreadStep>();
90        classPool = EngineeringUtil.createClassPool(classLoader);
91        threadStepCtClass = classPool.get("org.xchain.framework.lifecycle.ThreadStep");
92  
93        super.scan();
94  
95        try {
96          List<QName> threadStepQNameList = dependencySorter.sort();
97          for( QName threadStepQName : threadStepQNameList ) {
98            ThreadStep threadStep = threadStepMap.get(threadStepQName);
99            if( threadStep != null ) {
100             threadStepList.add(threadStep);
101           }
102         }
103       }
104       catch( DependencyCycleException dce ) {
105         throw new ScanException("There is at least one cycle in the thread step dependencies.", dce);
106       }
107     }
108     catch( NotFoundException nfe ) {
109       throw new ScanException("The definition of a required class could not be found.", nfe);
110     }
111     catch( Exception e ) {
112       throw new ScanException("An unknown exception was thrown from the thread step scanner.", e);
113     }
114     finally {
115       dependencySorter = null;
116       classPool = null;
117       threadStepCtClass = null;
118     }
119   }

131   public void scanNode( ScanNode node )
132     throws ScanException
133   {
134     try {
135       if( isLoadableClassFile(node.getResourceName()) ) {
136         String className = toClassName(node);
137 
138         CtClass scannedCtClass = null;
139         boolean requiresAnnotationScanning = false;
140 
141         try {
142           scannedCtClass = classPool.get(className);
143           requiresAnnotationScanning = hasAnnotation(scannedCtClass, LifecycleClass.class);
144         }
145         catch( Exception e ) {
146           if( log.isDebugEnabled() ) {
147             log.debug("Could not scan file '"+node.getResourceName()+"' due to an exception.", e);
148           }
149           return;
150         }
151         if( requiresAnnotationScanning ) {
152           // get the actual class.
153           Class lifecycleClass = context.getClassLoader().loadClass(className);
154 
155           // get the lifecycle annotation.
156           LifecycleClass lifecycleAnnotation = (LifecycleClass)lifecycleClass.getAnnotation(LifecycleClass.class);
157           
158           // get the uri for the annotations found in the class.
159           String namespaceUri = lifecycleAnnotation.uri();
160 
161           // scan the lifecycle class for a lifecycle accessor method.
162           Method accessorMethod = null;
163           int accessorMethodCount = 0;
164           for( Method method : lifecycleClass.getDeclaredMethods() ) {
165             if( Modifier.isStatic(method.getModifiers()) && hasAnnotation(method, LifecycleAccessor.class) ) {
166               accessorMethod = method;
167               accessorMethodCount++;
168             }
169           }
170 
171           if( accessorMethodCount > 1 ) {
172             throw new ScanException("The class '"+lifecycleClass+"' has "+accessorMethodCount+" static methods that are annotated with the LifecycleAccessor annotation."+
173                      " Lifecycle classes should have at most one accessor method.");
174           }
175 
176           // scan the class for for start and stop steps.
177           for( Method method : lifecycleClass.getMethods() ) {
178             boolean isStartStep = hasAnnotation(method, StartThreadStep.class);
179             boolean isStopStep = hasAnnotation(method, StopThreadStep.class);
180             boolean isStatic = Modifier.isStatic(method.getModifiers());
181 
182             if( isStartStep && isStopStep ) {
183               throw new ScanException("The class '"+lifecycleClass+"' has a lifecycle method '"+method.getName()+"' that has both a StartStep and a StopStep annotations.");
184             }
185 
186             // if this is not a start or stop step, then move on.
187             if( !isStartStep && !isStopStep ) {
188               continue;
189             }
190             
191             // ASSERT: The method has a single step annotation.
192 
193             assertProperStepMethodSignature(lifecycleClass, method);
194 
195             // ASSERT: The method has the proper signature.
196 
197             String localName = null;
198             QName qName = null;
199             Set<QName> beforeSet = null;
200             Set<QName> afterSet = null;
201 
202             if( isStartStep ) {
203               // get the start step annotation.
204               StartThreadStep startStep = method.getAnnotation(StartThreadStep.class);
205               localName = startStep.localName();
206               qName = new QName(namespaceUri, localName);
207               beforeSet = toQNameSet(startStep.before(), namespaceUri);
208               afterSet = toQNameSet(startStep.after(), namespaceUri);
209             }
210             else if( isStopStep ) {
211               StopThreadStep stopStep = method.getAnnotation(StopThreadStep.class);
212               localName = stopStep.localName();
213               qName = new QName(namespaceUri, localName);
214               afterSet = toQNameSet(stopStep.before(), namespaceUri);
215               beforeSet = toQNameSet(stopStep.after(), namespaceUri);
216             }
217             
218             // ASSERT: the qName, beforeSet, and afterSet have all been set from the perspective of a start step.
219             
220             if( isStartStep ) {
221               // get the start step annotation.
222               StartThreadStep startStep = method.getAnnotation(StartThreadStep.class);
223             }
224             
225             // ASSERT: prefix mappings set for the start step.
226 
227             ThreadStep step = threadStepMap.get(qName);
228             if( step == null ) {
229               step = new AnnotationThreadStep(qName);
230             }
231 
232             AnnotationThreadStep annotationStep = (AnnotationThreadStep)step;
233 
234             // set the accessor method on the annotation step.
235             annotationStep.setLifecycleAccessor(accessorMethod);
236 
237             // make sure that the annotation step does not already have a step defined.
238             if( isStartStep && annotationStep.getStartMethod() != null ) {
239               throw new ScanException("The class '"+lifecycleClass+"' has more than one start step for QName '"+qName+"'.");
240             }
241             else if( isStopStep && annotationStep.getStopMethod() != null ) {
242               throw new ScanException("The class '"+lifecycleClass+"' has more than one stop step for QName '"+qName+"'.");
243             }
244 
245             // ASSERT: There is not a duplicate start or stop step for this qName.
246 
247             if( isStartStep ) {
248               annotationStep.setStartMethod(method);
249             }
250             else if (isStopStep) {
251               annotationStep.setStopMethod(method);
252             }
253 
254             // add the lifecycle step to the map.
255             threadStepMap.put(qName, annotationStep);
256 
257             // add information about the step to the sorter.
258             dependencySorter.add(qName);
259             for( QName before : beforeSet ) {
260               dependencySorter.addDependency(qName, before);
261             }
262             for( QName after : afterSet ) {
263               dependencySorter.addDependency(after, qName);
264             }
265           }
266            
267         }
268       }
269     }
270     catch( Exception e ) {
271       throw new ScanException("Failed to scan file '"+node.getResourceName()+"' due to an exception.", e);
272     }
273   }

275   private static void assertProperStepMethodSignature(Class<?> lifecycleClass, Method method) throws ScanException {
276     // check the parameters on the method, to make sure it is a valid method signature.
277     Class<?>[] parameterTypes = method.getParameterTypes();
278     
279     // make sure that the parameter list is valid.  If it isn't, tell the user about the problem.
280     if( parameterTypes.length == 1 && (LifecycleContext.class.isAssignableFrom(parameterTypes[0])) ) {
281       throw new ScanException("The class '"+lifecycleClass+"' has a thread lifecycle method '"+method.getName()+"' that has an illegal signature."+
282                               " "+VALID_STEP_METHOD_SIGNATURES);
283     }
284     if( parameterTypes.length > 1 ) {
285       throw new ScanException("The class '"+lifecycleClass+"' has a thread lifecycle method '"+method.getName()+"' that has an illegal signature."+
286           " "+VALID_STEP_METHOD_SIGNATURES);
287     }
288   }

290   private static Set<QName> toQNameSet( String[] qNameArray, String defaultUri )
291     throws ScanException
292   {
293     Set<QName> qNameSet = new HashSet<QName>();
294     for( String qNameString : qNameArray ) {
295       QName qName = null;
296       if( qNameString.startsWith("{") ) {
297         qName = QName.valueOf(qNameString);
298       }
299       else if( qNameString.matches("[A-Za-z][-A-Za-z0-9._]*") ) {
300         qName = new QName( defaultUri, qNameString );
301       }
302       else {
303         throw new ScanException("The qname string '"+qNameString+"' does not appear to be a valid qname or local name.");
304       }
305       qNameSet.add(qName);
306     }
307     return qNameSet;
308   }
