114   public static void startLifecycle( LifecycleContext context )
115     throws LifecycleException
116   {
117     synchronized( XmlFactoryLifecycle.class ) {
118       started = true;
119 
120       // log all of the factories that have been added to this lifecycle.
121       if( log.isInfoEnabled() ) {
122         StringBuilder status = new StringBuilder();
123         status.append("XML Lifecycle State\n");
124         if( saxParserFactoryMap.isEmpty() ) {
125           status.append("  There are no Factory<SAXParserFactory> objects defined.\n");
126         }
127         else {
128           status.append("  SAXParserFactories:\n");
129           for( Map.Entry<QName, Factory<SAXParserFactory>> entry : saxParserFactoryMap.entrySet() ) {
130             status.append("    ").append(entry.getKey()).append(" => ").append(entry.getValue().getClass().getName()).append("\n");
131           }
132         }
133         if( transformerFactoryMap.isEmpty() ) {
134           status.append("  There are no Factory<TransfromerFactory> objects defined.\n");
135         }
136         else {
137           status.append("  SAXTransformerFactories:\n");
138           for( Map.Entry<QName, Factory<SAXTransformerFactory>> entry : transformerFactoryMap.entrySet() ) {
139             status.append("    ").append(entry.getKey()).append(" => ").append(entry.getValue().getClass().getName()).append("\n");
140           }
141         }
142         log.info(status.toString());
143       }
144     }
145   }

152   /**
153    * Puts a new SAXParserFactory factory into the XmlFactoryLifecycle with the specified name.  If a factory is already bound to the
154    * specified name, it is removed.
155    */
156   public static void putSaxParserFactoryFactory( QName name, Factory<SAXParserFactory> factory )
157   {
158     synchronized( XmlFactoryLifecycle.class ) {
159       if( started ) {
160         throw new IllegalStateException("You may not add factories to the sax parser factory while the xml lifecycle is on.");
161       }
162       synchronized( saxParserFactoryMap ) {
163         saxParserFactoryMap.put( name, factory );
164       }
165     }

281   {
282     Factory<SAXTransformerFactory> factory = getTransformerFactoryFactory( name );

285       throw new RuntimeException("There is no Factory<SAXTransformerFactory> named '"+name+".");
286     }
287

291   public static SAXTransformerFactory newTransformerFactory( String uri, String localName, Attributes attributes )
292   {
293     // TODO: replace this with some kind of configurable strategy for looking up the local name.
294     // if the uri is equal to the xslt uri, then use the default xslt factory.
295     if( XSLT_NAMESPACE.equals(uri) ) {
296       return newTransformerFactory(SAXON_FACTORY_NAME);
297     }
298     else if( STX_NAMESPACE.equals(uri) ) {
299       return newTransformerFactory(JOOST_FACTORY_NAME);
300     }
301     else {
302       throw new RuntimeException("There is no Factory<SAXTransformerFactory> for '"+uri+"', '"+localName+"'.");
303     }
304   }
