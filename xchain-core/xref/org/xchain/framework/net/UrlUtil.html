88    /**
89     * Check if the given URL exists.  This uses UrlExistsStrategy implementations to determine if the given URL exists.
90     * 
91     * @param url The URL to check.
92     * 
93     * @return True if the given URL exists.
94     */
95    public boolean exists( URL url )
96      throws Exception
97    {
98      // get the protocol.
99      String protocol = url.getProtocol();
100 
101     // get the strategy for this protocol.
102     UrlExistsStrategy strategy = (UrlExistsStrategy)existsStrategyMap.get(protocol);
103 
104     if( log.isDebugEnabled() ) {
105       log.debug("Checking if '"+url.toExternalForm()+"' is exists ("+ protocol+").");
106     }
107 
108     if( strategy != null ) {
109       return strategy.exists(url);
110     }
111     else {
112       URLConnection connection = null;
113       try {
114         connection = url.openConnection();
115         connection.connect();
116         String statusCode = connection.getHeaderField("Status-Code");
117         if( statusCode == null ) {
118           if( log.isDebugEnabled() ) {
119             log.debug("The status code for '"+url.toExternalForm()+"' is null.");
120           }
121           return false;
122         }
123         else {
124           return statusCode.matches("2\\d\\d");
125         }
126       }
127       catch (ResourceNotFoundException rnfe) {
128         return false;
129       }
130       finally {
131         closeConnection(connection);
132       }
133     }
134   }

153   /**
154    * Returns the last modified date for this url connection.
155    */
156   public long lastModified( URLConnection connection )
157   {
158     long lastModified = 0;
159 
160     // if this url connection is a jar url, then we need to try and get the date from the entry.
161     try {
162       if( connection instanceof JarURLConnection ) {
163         try {
164           JarURLConnection jarConnection = (JarURLConnection)connection;
165           JarEntry jarEntry = jarConnection.getJarEntry();
166           if( jarEntry != null ) {
167             lastModified = jarEntry.getTime();
168           }
169         }
170         catch( Exception e ) {
171           if( log.isWarnEnabled() ) {
172             log.warn("Could not get date from jar entry.", e);
173           }
174         }
175       }
176 
177       // if the last modified date was not set, then return the standard last modified date.
178       if( lastModified <= 0 ) {
179         lastModified = connection.getLastModified();
180       }
181     }
182     finally {
183       closeConnection(connection);
184     }
185 
186     return lastModified;
187   }

234   /**
235    * Determine if the given url was last modified before the given set of urls.
236    * 
237    * @param url The url to check with.
238    * @param urlSet The set of urls to check against.
239    * 
240    * @return True if the given url was last modified before any of the urls in the given set.
241    */
242   public boolean lastModifiedBefore( URL url, Set<URL> urlSet )
243     throws IOException
244   {
245     boolean result = false;
246     URLConnection connection = null;
247     URLConnection otherConnection = null;
248     
249     try {
250       // Open the connection.
251       connection = url.openConnection();
252       // Get the last modified date.
253       long lastModified = lastModified(connection);
254       if( log.isDebugEnabled() ) {
255         log.debug("Url '"+url+"' has last modified date "+lastModified+".");
256       }
257   
258       // Iterate over the given set of URLs.
259       Iterator<URL> urlIterator = urlSet.iterator();
260       while( urlIterator.hasNext() && !result ) {
261         URL otherUrl = urlIterator.next();
262         // Open the other connection.
263         otherConnection = otherUrl.openConnection();
264         // Get the last modified date.
265         long otherLastModified = lastModified(otherConnection);
266         if( log.isDebugEnabled() ) {
267           log.debug("Url '"+otherUrl+"' has last modified date "+otherLastModified+".");
268         }
269         // Close the other connection.
270         closeConnection(otherConnection);
271         result = lastModified < otherLastModified;
272       }
273     } finally {
274       // Make sure both connections are closed.
275       closeConnection(connection);
276       closeConnection(otherConnection);
277     }
278 
279     return result;
280   }

282   /**
283    * Check if the given URL has been modified after the given date.
284    * 
285    * @param url The URL to check.
286    * @param lastModified The last modified date to check on.
287    * 
288    * @return True if the given URL has been modified after the given date.

290   public boolean lastModifiedAfter( URL url, long lastModified )
291     throws IOException
292   {
293     // get the load time for the url.
294     long urlLastModified = lastModified(url);
295 
296     // if the url last modified is after the load date, then a reload is required.
297     return lastModified < urlLastModified;

301   /**
302    * Check if the given collection of URLs has been modified after the given date.
303    * 
304    * @param urlCollection The collection of URLs to check.
305    * @param lastModified The last modified date to check on.
306    * 
307    * @return True if any of the given URLs has been modified after the given date.
308    */
309   public boolean lastModifiedAfter( Collection<URL> urlCollection, long lastModified )
310     throws IOException
311   {
312     boolean result = false;
313 
314     // if the load date is before any of the urls in the dependency set, then we need to reload.
315     Iterator<URL> urlIterator = urlCollection.iterator();
316     while( urlIterator.hasNext() && !result ) {
317       result = lastModifiedAfter(urlIterator.next(), lastModified);
318     }
319 
320     return result;
321   }

354   public static void closeInputStream( URLConnection urlConnection )
355   {
356     if( urlConnection != null && urlConnection.getDoInput() ) {
357       try {
358         urlConnection.getInputStream().close();
