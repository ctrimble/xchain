107   /**
108    * This method parses the Bundle-ClassPath manifest header.
109    *
110    * Bundle-ClassPath BNF: (OSGi 4.2 Specification - 3.8.1)
111    *   Bundle-ClassPath::=  entry ( ’,’ entry )* 
112    *   entry ::= target ( ’;’ target )* ( ’;’ parameter ) * 
113    *   target ::= path | ’.’
114    *
115    * Path BNF: (OSGi 4.2 Specification - 1.3.2)
116    *   path ::= path-unquoted | (’"’ path-unquoted ’"’) 
117    *   path-unquoted::= path-sep | path-sep? path-element (path-sep path-element)* 
118    *   path-element ::= [^/"\#x0D#x0A#x00]+ 
119    *   path-sep ::= ’/’
120    */
121   public static List<ParsedClassPathEntry> parseClassPathEntries( String bundleClassPath )
122     throws Exception
123   {
124     List<ParsedClassPathEntry> entryList = new ArrayList<ParsedClassPathEntry>();
125 
126     // create the matcher.
127     Matcher matcher = ZERO_OR_MORE_WHITESPACE_PATTERN.matcher(bundleClassPath);
128 
129     // remove leading whitespace.
130     consumeWhitespace(matcher, false);
131 
132     // get the first entry.
133     ParsedClassPathEntry entry = parseClassPathEntry(matcher);
134     entryList.add(entry);
135     consumeWhitespace(matcher, false);
136 
137     while( lookingAt(matcher, COMMA_PATTERN) ) {
138       advanceRegion(matcher);
139       consumeWhitespace(matcher, false);
140       entry = parseClassPathEntry(matcher);
141       entryList.add(entry);
142       consumeWhitespace(matcher, false);
143     }
144 
145     consumeWhitespace(matcher, false);
146 
147     // verify that we are at the end of the matcher.
148     // TODO: Make sure that we consumed all of the header.
149 
150     return entryList;
151   }

153   static ParsedClassPathEntry parseClassPathEntry( Matcher matcher )
154     throws Exception
155   {
156     ParsedClassPathEntry entry = new ParsedClassPathEntry();
157 
158     Pattern originalPattern = matcher.pattern();
159 
160     try {
161       entry.getTargetList().add(parseTarget(matcher));
162       consumeWhitespace( matcher, false );
163 
164       // some lookahead is needed here to properly parse this part of the grammar.
165       while( lookingAt(matcher, SEMICOLON_PATTERN) ) {
166         int regionStart = matcher.regionStart();
167         advanceRegion( matcher );
168         consumeWhitespace( matcher, false );
169 
170         // if this could not be a parameter, then consume the token.
171         if( !lookingAt(matcher, START_PARAMETER_PATTERN) ) {
172           entry.getTargetList().add(parseTarget(matcher));
173           consumeWhitespace( matcher, false );
174         }
175         else {
176           // reset the region, so that we can parse the parameters.
177           matcher.region(regionStart, matcher.regionEnd());
178         }
179       }
180 
181       while( lookingAt(matcher, SEMICOLON_PATTERN) ) {
182         advanceRegion( matcher );
183         consumeWhitespace( matcher, false );
184         entry.getParameterList().add(parseParameter(matcher));
185       }
186     }
187     finally {
188       matcher.usePattern(originalPattern);
189     }
190 
191     return entry;
192   }

194   /**
195    *   target ::= path | ’.’
196    *   path ::= path-unquoted | (’"’ path-unquoted ’"’) 
197    *   path-unquoted::= path-sep | path-sep? path-element (path-sep path-element)* 
198    *   path-element ::= [^/"\#x0D#x0A#x00]+ 
199    *   path-sep ::= ’/’
200    */
201   static String parseTarget( Matcher matcher )
202     throws Exception
203   {
204     Pattern originalPattern = matcher.pattern();
205 
206     try {
207       if( lookingAt(matcher, QUOTE_PATTERN) ) {
208         return parseQuotedPath(matcher);
209       }
210       else {
211         return parseUnquotedPath(matcher);
212       }
213     }
214     finally {
215       matcher.usePattern(originalPattern);
216     }
217   }

219   static String parseQuotedPath( Matcher matcher )
220     throws Exception
221   {
222     Pattern originalPattern = matcher.pattern();
223     try {
224       if( lookingAt( matcher, QUOTED_PATH_PATTERN ) ) {
225         String path = matcher.group(1);
226         advanceRegion( matcher );
227         return path;
228       }
229       else {
230         throw new RuntimeException("Could not parse quoted path.");
231       }
232     }
233     finally {
234       matcher.usePattern(originalPattern);
235     }
236   }

238   static String parseUnquotedPath( Matcher matcher )
239     throws Exception
240   {
241     Pattern originalPattern = matcher.pattern();
242     try {
243       if( lookingAt( matcher, UNQUOTED_PATH_PATTERN ) ) {
244         String path = matcher.group().trim();
245         advanceRegion( matcher );
246         return path;
247       }
248       else {
249         throw new RuntimeException("Could not parse quoted path.");
250       }
251     }
252     finally {

257   /**
258    * Parses a parameter from the specified matcher.  The matcher passed to this method must be looking at the start of a parameter.
259    *
260    *   parameter ::= directive | attribute 
261    *   directive ::= token ’:=’ argument 
262    *   attribute ::= token ’=’ argument
263    *
264    * @param matcher the matcher to parse the parameter from.
265    * @return the parsed parameter.
266    */
267   static ParsedParameter parseParameter( Matcher matcher )
268     throws Exception
269   {
270     ParsedParameter parameter = new ParsedParameter();
271 
272     Pattern originalPattern = matcher.pattern();
273     try {
274       if( lookingAt(matcher, TOKEN_PATTERN) ) {
275         parameter.setName(matcher.group());
276         advanceRegion(matcher);
277       }
278       else {
279         throw new RuntimeException("Could not find token at start of parameter.");
280       }
281 
282       consumeWhitespace(matcher, false);
283 
284       if( lookingAt( matcher, COLON_EQUALS_PATTERN ) ) {
285         parameter.setType(ParameterType.DIRECTIVE);
286         advanceRegion(matcher);
287       }
288       else if( lookingAt( matcher, EQUALS_PATTERN ) ) {
289         parameter.setType(ParameterType.ATTRIBUTE);
290         advanceRegion(matcher);
291       }
292       else {
293         throw new RuntimeException("Expecting := or =");
294       }
295 
296       consumeWhitespace(matcher, false);
297 
298       parameter.setValue(parseArgument(matcher));
299     }
300     finally {
301       matcher.usePattern(originalPattern);
302     }
303 
304     return parameter;
305   }

307   /**
308    * @param matcher the matcher that the whitespace will be consumed from.
309    * @param mandatory if true, the whitespace in this location mandatory, otherwise
310    *                   the whitespace is optional.
311    */
312   static void consumeWhitespace( Matcher matcher, boolean mandatory )
313     throws Exception
314   {
315     Pattern originalPattern = matcher.pattern();
316 
317     try {
318       if( lookingAt( matcher, mandatory ? ONE_OR_MORE_WHITESPACE_PATTERN : ZERO_OR_MORE_WHITESPACE_PATTERN ) ) {
319         advanceRegion(matcher);
320       }
321       else {
322         throw new RuntimeException("Expected whitespace, but there was none.");
323       }
324     }
325     finally {
326       matcher.usePattern(originalPattern);
327     }
328   }

330   /**
331    * Perses an argument from the OSGi Core Specification.  Any quoted string are unescaped by this method.
332    *
333    *   extended ::= ( alphanum | ’_’ | ’-’ | ’.’ )+ 
334    *   quoted-string::= ’"’ ( [^"\#x0D#x0A#x00] | ’\"’|’\\’)* ’"’
335    *   argument ::= extended  | quoted-string 
336    *
337    */
338   static String parseArgument( Matcher matcher )
339     throws Exception
340   {
341     Pattern originalPattern = matcher.pattern();
342 
343     try {
344       if( lookingAt( matcher, EXTENDED_PATTERN ) ) {
345         return matcher.group();
346       }
347       else if( lookingAt( matcher, QUOTED_STRING_PATTERN ) ) {
348         return unescapeQuotedString(matcher.group());
349       }
350       // TODO: Better error handling here.
351       throw new RuntimeException("Could not find argument.");
352     }
353     finally {
354       matcher.usePattern(originalPattern);
355     }
356   }

358   /**
359    * Removed the surrounding quotation marks and unescapes '"' and '\' characters in a quoted string.  This method assumes
360    * that the string passed in conforms to the definition of a quoted-string found in the osgi core specification.  If the string
361    * is malformed, the results of this method are unspecified.
362    *
363    * @param quotedString a string that conforms the quoted-string BNF found in the OSGi Core Specification.
364    * @return the unescaped string.
365    */
366   static String unescapeQuotedString(String quotedString)
367   {
368     return quotedString.replaceAll("\\A\"(.*)\"\\Z", "$1").replaceAll("\\\\([\\\\\"])", "$1");
369   }

371   /**
372    *   digit ::= [0..9] 
373    *   alpha ::= [a..zA..Z] 
374    *   alphanum ::= alpha | digit 
375    *   token ::= ( alphanum | ’_’ | ’-’ )+
376    */
377   static String parseToken( Matcher matcher )
378     throws ParseException
379   {
380     Pattern originalPattern = matcher.pattern();
381 
382     try {
383       matcher.usePattern(TOKEN_PATTERN);
384 
385       if( matcher.lookingAt() ) {
386         return matcher.group();
387       }
388       else {
389         throw new RuntimeException("Could not parse token at "+matcher.regionStart());
390       }
391     }
392     finally {
393       matcher.usePattern(originalPattern);
394     }
395   }

397   /**
398    * Returns true if the matcher is looking at the specified pattern, false otherwise.  If true is returned, the pattern for the matcher is the pattern specified.
399    * If false is returned, then the pattern for the matcher is not changed.
400    * NOTE: This should be moved to the RegexUtil.
401    */
402   static boolean lookingAt( Matcher matcher, Pattern pattern )
403     throws Exception
404   {
405     Pattern originalPattern = matcher.pattern();
406     boolean lookingAt = false;
407 
408     try {
409       matcher.usePattern(pattern);
410       return (lookingAt = matcher.lookingAt());
411     }
412     finally {
413       if( !lookingAt ) {
414         matcher.usePattern(originalPattern);
415       }
416     }
