44  /**
45   * This handler provides caching and translation services for command based sax sources.  The following features are implemented:
46   *
47   * 1) Attributes can be assigned after a startElement event by calling attribute( String namespace, String localName, String qName );
48   * 2) The handler can be set into comment mode, which will translate all calls into comments.
49   * 3) The handler can be flushed, to force all outstanding events to the content handler.  This will be a partial flush when in comment mode, since
50   * comments are passed in one call, instead of many.
51  
52   * Modes:
53   * 1) Standard mode.
54   * 2) Element building mode.
55   *   2.1) Attribute building mode.
56   * 3) Comment building mode.
57   *
58   * Element building is terminated by:
59   * 1) A text event (characters or ignorable whitespace.)
60   * 2) Another start element event.
61   * 3) A processing instruction.
62   * 4) An end element event.
63   *
64   * Attribute building is terminated by:
65   * 1) An end attribute event.
66   *
67   * Comment building is terminated by:
68   * 1) an end comment event.
69   *
70   * @author Mike Moulton
71   * @author Devon Tackett
72   * @author Christian Trimble
73   * @author Jason Rose
74   * @author Josh Kennedy
75   */

205   /**
206    * Sets up a start element event that will be sent to the content handler.  When the simple handler detects that the
207    * creation of an element is complete, then start prefix mapping events and a start element event will be fired.
208    */
209   public void startElement(String namespace, String localName, String qName, Attributes attributes)
210     throws SAXException
211   {
212     if( !redirectCharacters() ) {
213       fireOutstandingEvents();
214 
215       // CHANGE: push the next element.
216       // add the prefix mappings that need to be added to this element.
217       // clear the current prefix mappings.
218 
219       // update the input namespace context.
220       for( Map.Entry<String, String> prefixMapping : nextPrefixMapping.entrySet() ) {
221         inputNamespaceContext.startPrefixMapping(prefixMapping.getKey(), prefixMapping.getValue());
222       }
223 
224       nextElement = new Element(namespace, localName, qName);
225 
226       // copy all of the prefix mappings that were found for this element.
227       nextElement.getPrefixMappings().putAll(nextPrefixMapping);
228       nextPrefixMapping.clear();
229 
230       for( int i = 0; i < attributes.getLength(); i++ ) {
231         nextElement.addAttribute( attributes.getURI(i), attributes.getLocalName(i), attributes.getQName(i), attributes.getValue(i) );
232       }
233 
234       elementStack.addFirst(nextElement);
235     }
236   }

239    * Sends the end element event to the content handler.
240    */
241   public void endElement(String namespace, String localName, String qName)
242     throws SAXException
243   {
244     if( !redirectCharacters() ) {
245 
246       // fires outstanding events.
247       fireOutstandingEvents();
248 
249       // remove the first element from the stack.
250       Element element = elementStack.removeFirst();
251 
252       // make sure that this is the correct element.
253       // TODO: add sanity checks.
254 
255       // end the element.
256       contentHandler().endElement(element.getNamespace(), element.getLocalName(), element.getQName());
257 
258       for( String prefix : element.getPrefixMappings().keySet() ) {
259         contentHandler().endPrefixMapping(prefix);
260         outputNamespaceContext.endPrefixMapping(prefix);
261       }
262     }

265   public void startAttribute( String namespace, String localName, String qName )
266     throws SAXException
267   {
268     if( !redirectCharacters() ) {
269       startRedirectCharacters();
270     }
271 
272     redirectDepth++;
273 
274     if( redirectDepth == 1 && nextElement != null ) {
275       // add any missing prefix mappings to the next element.
276       for( Map.Entry<String, String> prefixMapping : nextPrefixMapping.entrySet() ) {
277         String currentMapping = nextElement.getPrefixMappings().get(prefixMapping.getKey());
278         if( currentMapping == null ) {
279           currentMapping = outputNamespaceContext.lookUpNamespaceUri(prefixMapping.getKey());
280         }
281         if( currentMapping == null || currentMapping.equals(prefixMapping.getValue()) ) {
282           inputNamespaceContext.startPrefixMapping(prefixMapping.getKey(), prefixMapping.getValue());
283           elementStack.getFirst().getPrefixMappings().put(prefixMapping.getKey(), prefixMapping.getValue());
284         }
285         else {
286           throw new SAXException("Cannot map prefix '"+prefixMapping.getKey()+"' to '"+prefixMapping.getValue()+"' because it is already mapped to '"+currentMapping+"'.");
287         }
288       }
289     }
290     else {
291       for( Map.Entry<String, String> prefixMapping : nextPrefixMapping.entrySet() ) {
292         inputNamespaceContext.startPrefixMapping(prefixMapping.getKey(), prefixMapping.getValue());
293       }
294     }
295     nextPrefixMapping.clear();
296   }

339   public void startExcludeResultPrefix( String prefix )
340     throws SAXException
341   {
342     String namespace = null;
343 
344     // try to get the namespace from the next prefix mapping, since this happens before outstanding events
345     // are fired.
346     if( "#all".equals(prefix) ) {
347       namespace = "#all";
348     } 
349     else if( nextPrefixMapping.containsKey(prefix) ) {
350       namespace = nextPrefixMapping.get(prefix);
351     }
352     else {
353       namespace = inputNamespaceContext.lookUpNamespaceUri(prefix);
354     }
355 
356     if( namespace == null ) {
357       throw new SAXException("The prefix '"+prefix+"' was excluded, but is not defined in the namespace context.");
358     }
359 
360     excludeNamespaceContextStack.getFirst().startPrefixMapping(prefix, namespace);
361   }

532   /**
533    * Fires any element events that have been cached by this content handler.
534    */
535   private void fireOutstandingEvents()
536     throws SAXException
537   {
538     if( log.isDebugEnabled() ) {
539       log.debug("fireOutstandingEvents() called.");
540     }
541 
542     // if there is a next element.
543     if( nextElement != null ) {
544 
545       // the set of all of the namespaces that are required by this element.
546       Map<String, String> requiredPrefixMappings = new HashMap<String, String>();
547       requiredPrefixMappings.put(parsePrefix(nextElement.getQName()), nextElement.getNamespace());
548 
549       // create the set of attributes for this element.
550       AttributesImpl attributes = new AttributesImpl();
551       Iterator<Attribute> attributeIterator = nextElement.getAttributeMap().values().iterator();
552       while( attributeIterator.hasNext() ) {
553         Attribute attribute = (Attribute)attributeIterator.next();
554 
555         attributes.addAttribute(attribute.getNamespace(), attribute.getLocalName(), attribute.getQName(), "CDATA", attribute.getValue());
556         if( attribute.getNamespace() != null && !"".equals(attribute.getNamespace()) ) {
557           requiredPrefixMappings.put(parsePrefix(attribute.getQName()), attribute.getNamespace());
558         }
559       }
560 
561       // ASSERT: the attributes for this element are all constructed.
562       // ASSERT: we know all of the required prefix mappings for this element.
563       // ASSERT: the requiredPrefixMappings is a subset of nextElement.getPrefixMappings()
564 
565       // remove the required mappings from the nextElement, we will put them back when the trimming of the prefixes is done.
566       nextElement.getPrefixMappings().keySet().removeAll(requiredPrefixMappings.keySet());
567 
568       // remove all of the mappings that will have no effect on the output document.
569       removeUnneededMappings(outputNamespaceContext, nextElement.getPrefixMappings());
570       removeUnneededMappings(outputNamespaceContext, requiredPrefixMappings);
571 
572       // remove all excluded prefix mappings from the namespace mappings left in nextElement.getPrefixMappings().  These are the namespaces that are
573       // not required right now.
574       if( !excludeNamespaceContextStack.isEmpty() ) {
575         PrefixMappingContext excludeResultContext = excludeNamespaceContextStack.getFirst();
576         if( excludeResultContext.contains("#all", "#all") ) {
577           nextElement.getPrefixMappings().clear();
578         } 
579         else { 
580           Iterator<Map.Entry<String, String>> mappingIterator = nextElement.getPrefixMappings().entrySet().iterator();
581           while( mappingIterator.hasNext() ) {
582             Map.Entry<String, String> entry = mappingIterator.next();
583             if( excludeResultContext.contains(entry.getKey(), entry.getValue()) ) {
584               mappingIterator.remove();
585             }
586           }
587         }
588       }
589 
590       // add all of the required prefix mappings back.
591       nextElement.getPrefixMappings().putAll(requiredPrefixMappings);
592 
593       // ASSERT: all prefixes that must be output by this element are now in nextElement.getPrefixMappings().
594 
595       // output all of the namespace prefixes.
596       for( Map.Entry<String, String> prefixMapping : nextElement.getPrefixMappings().entrySet() ) {
597         contentHandler().startPrefixMapping(prefixMapping.getKey(), prefixMapping.getValue());
598         outputNamespaceContext.startPrefixMapping(prefixMapping.getKey(), prefixMapping.getValue());
599       }
600 
601       // ASSERT: the prefix mappings are now all defined for the context of this element.
602 
603       // fire the start event for the element.
604       contentHandler().startElement(nextElement.getNamespace(), nextElement.getLocalName(), nextElement.getQName(), attributes );
605 
606       // ASSERT: the start element has been fired for this element.
607 
608       if( log.isDebugEnabled() ) {
609         log.debug("Adding element '"+nextElement.getQName()+"' to the node stack.");
610       }
611 
612       // clear the next element.
613       nextElement = null;
614     }
615   }

617   /**
618    * Removes mappings from the map prefixMappings that are already defined in prefixMappingContext.
619    *
620    * @param prefixMappingContext the prefix mapping context that will be checked for mappings.
621    * @param prefixMappings the prefix map that will have unneeded mappings removed.
622    */
623   private static void removeUnneededMappings( PrefixMappingContext prefixMappingContext, Map<String, String> prefixMappings )
624   {
625     Iterator<Map.Entry<String, String>> prefixMappingIterator = prefixMappings.entrySet().iterator();
626     while( prefixMappingIterator.hasNext() ) {
627       Map.Entry<String, String> prefixMappingEntry = prefixMappingIterator.next();
628       if( prefixMappingContext.contains(prefixMappingEntry.getKey(), prefixMappingEntry.getValue()) ) {
629         prefixMappingIterator.remove();
630       }
631     }
632   }

634   /**
635    * Parses a prefix from a QName.  If the qName does not have a prefix, then "" is returned, otherwise the prefix portion of the QName is returned.
636    *
637    * @param qName the QName to parse.
638    * @return the prefix part of the QName, or the empty string if there is not a prefix in the QName.
