16  package org.xchain.framework.sax;

18  import java.util.Iterator;
19  import org.xml.sax.Attributes;
20  import static org.junit.Assert.*;
21  
22  /**
23   * @author Christian Trimble
24   */
25  public class SaxEventRecorderAssert
26  {
27    public static void assertStartDocument(Iterator<SaxEventRecorder.SaxEvent> eventIterator)
28      throws Exception
29    {
30      assertTrue("There is not a start document event", eventIterator.hasNext());
31      SaxEventRecorder.SaxEvent event = eventIterator.next();
32      assertEquals("There was not a start document event.", SaxEventRecorder.EventType.START_DOCUMENT, event.getType());
33    }
34  
35    public static void assertEndDocument(Iterator<SaxEventRecorder.SaxEvent> eventIterator)
36      throws Exception
37    {
38      assertTrue("There is not an end document event", eventIterator.hasNext());
39      SaxEventRecorder.SaxEvent event = eventIterator.next();
40      assertEquals("There was not an end document event.", SaxEventRecorder.EventType.END_DOCUMENT, event.getType());
41    }
42  
43    public static void assertStartElement(Iterator<SaxEventRecorder.SaxEvent> eventIterator, String uri, String localName, Attributes attributes)
44      throws Exception
45    {
46      assertTrue("There is not a start element event.", eventIterator.hasNext());
47      SaxEventRecorder.SaxEvent event = eventIterator.next();
48      assertEquals("There was not a start element event.", SaxEventRecorder.EventType.START_ELEMENT, event.getType());
49      assertEquals("The element local name is not correct.", localName, event.getLocalName());
50      assertEquals("The element uri is not correct.", uri, event.getUri());
51  
52      // only check the attributes if they are not null.
53      if( attributes != null ) {
54        assertEquals("The element has the wrong number of attributes.", attributes.getLength(), event.getAttributes().getLength());
55        for( int i = 0; i < event.getAttributes().getLength(); i++ ) {
56          int attributesIndex = attributes.getIndex(event.getAttributes().getURI(i), event.getAttributes().getLocalName(i));
57          assertTrue("The attribute does not contain the attribute {"+event.getAttributes().getURI(i)+"}"+event.getAttributes().getLocalName(i)+".", -1 != attributesIndex);
58          assertEquals("The attribute {"+event.getAttributes().getURI(i)+"}"+event.getAttributes().getLocalName(i)+" has the wrong value.", attributes.getValue(attributesIndex), event.getAttributes().getValue(i));
59        }
60      }
61    }
62  
63    public static void assertEndElement(Iterator<SaxEventRecorder.SaxEvent> eventIterator, String uri, String localName )
64      throws Exception
65    {
66      assertTrue("There is not an end element event.", eventIterator.hasNext());
67      SaxEventRecorder.SaxEvent event = eventIterator.next();
68      assertEquals("There was not an end element event.", SaxEventRecorder.EventType.END_ELEMENT, event.getType());
69      assertEquals("The element local name is not correct.", localName, event.getLocalName());
70      assertEquals("The element uri is not correct.", uri, event.getUri());
71    }
72  
73    public static void assertCharacters(Iterator<SaxEventRecorder.SaxEvent> eventIterator, String text )
74      throws Exception
75    {
76      assertTrue("There is not an end element event.", eventIterator.hasNext());
77      SaxEventRecorder.SaxEvent event = eventIterator.next();
78      assertEquals("There was not an end element event.", SaxEventRecorder.EventType.CHARACTERS, event.getType());
79      assertEquals("The characters element does not have the correct value.", text, event.getText());
80    }
81  
82    public static void assertNoMoreEvents(Iterator<SaxEventRecorder.SaxEvent> eventIterator)
83      throws Exception
84    {
85      assertFalse("There were extra document events.", eventIterator.hasNext());
86    }
87  }
