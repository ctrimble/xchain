66     /* Pattern to match path names with a *-INF directory at the base, such as WEB-INF, OSGI-INF, etc. */
67     private static final Pattern INF_DIR_SUBPATH_PATTERN = Pattern.compile("^[/\\\\]?[^/\\\\]+-INF[/\\\\].*$", Pattern.CASE_INSENSITIVE);
68     
69     /**
70      * Returns true if the given name represents a path to a class file that does not have an INF directory at its 
71      * base. For example, <code>isLoadableClassFile('org/my/package/name/Foo.class')</code> will return true but
72      * <code>isLoadableClassFile('org/my/package/name/Foo')</code> and <code>isLoadableClassFile('WEB-INF/classes/org/my/package/name/Foo.class')</code>
73      * will return false.
74      * 
75      * @param name
76      * @return
77      */
78     public static boolean isLoadableClassFile(String name) {
79       return name.endsWith(".class") && !INF_DIR_SUBPATH_PATTERN.matcher(name).matches();
80     }
81     
82     public static String toClassName(ScanNode scanNode)
83     {
84        return scanNode.getResourceName().replaceAll("\\A(.*)\\.class\\Z", "$1").replaceAll("[/\\\\]", ".");
85     }

87      protected void scan()
88        throws ScanException
89      {
90        ScannerLifecycle scanLifecycle = ScannerLifecycle.getInstance();
91        LinkedList<ScanNode> stack = new LinkedList<ScanNode>();
92        try {
93          stack.addFirst(scanLifecycle.scanNode(rootUrlLocator));
94        }
95        catch( Exception e ) {
96          throw new ScanException("Could not build the root scan node.", e);
97        }
98  
99        while( !stack.isEmpty() ) {
100         ScanNode scanNode = stack.removeFirst();
101         stack.addAll(0, scanNode.getChildMap().values());
102 
103         if( scanNode.isFile() ) {
104           scanNode(scanNode);
105         }
106         else {
107           if( log.isDebugEnabled() ) {
108             log.debug("Scan node "+scanNode.getResourceName()+" not handled, it is a directory.");
109           }
110         }
111       }
112    }
