51    /**
52     * <p>Scans the entries in a Weblogic Zip URL and adds all of the entries of that jar into the tree rooted at rootNode.  Missing
53     * directory nodes will be added.</p>
54     *
55     * @param rootNode the root node of the scan.
56     * @param jarUrl the url of the jar with entries to be added to the scan node.
57     */
58    public void scan( ScanNode rootNode, URL bundleUrl )
59      throws IOException
60    {
61      if( log.isDebugEnabled() ) {
62        log.debug("Bundle url "+bundleUrl.toExternalForm());
63      }
64  
65      // get the bundle for this class.  This is used to get other bundles.
66      Bundle scannerBundle = FrameworkUtil.getBundle(BundleProtocolScanner.class);
67      BundleContext scannerBundleContext = scannerBundle.getBundleContext();
68  
69      // get the class path entry for this bundle.
70      String classPathHeader = (String)scannerBundle.getHeaders().get("Bundle-ClassPath");
71  
72      // parse the header.
73      List<ParsedClassPathEntry> classPathEntryList = null;
74      try {
75        classPathEntryList = ManifestParser.parseClassPathEntries( classPathHeader );
76      }
77      catch( Exception e ) {
78        throw new IOException("Could not scan "+bundleUrl);
79      }
80  
81      for( ParsedClassPathEntry entry : classPathEntryList ) {
82        for( String target : entry.getTargetList() ) {
83  
84          if( target.endsWith(".jar") ) {
85            scanJar( rootNode, target, scannerBundle );
86          }
87          else {
88            scanDirectory( rootNode, target, scannerBundle );
89          }
90        }
91      }
92    }

94    public void scanJar( ScanNode scanNode, String path, Bundle bundle )
95      throws IOException
96    {
97      URL jarUrl = bundle.getEntry(path);
98      JarInputStream in = null;
99      try {
100       in = new JarInputStream(jarUrl.openStream());
101       JarEntry entry = null;
102       while( (entry = in.getNextJarEntry()) != null ) {
103         String resourceName = entry.getName();
104         if( entry.isDirectory() && !resourceName.endsWith("/") )
105           resourceName = resourceName + "/";
106         insertEntryNodes( scanNode, resourceName );
107       }
108     }
109     catch( Exception e ) {
110       e.printStackTrace();
111       log.warn(format("Error occurred while processing JAR '%s'.", path), e);
112     }
113 
114   }

116   /**
117    * <p>Adds an entry to the root node for a JarEntry.  This method is declared to be protected, so that test cases an interact with it.
118    * In general, this method should be treaded as private.</p>
119    *
120    * @param rootNode the root node of the scan.
121    * @param entry the jar entry to place under the root node.
122    */
123   void insertEntryNodes( ScanNode rootNode, String name )
124   {
125     // TODO: verify that this works on windows.
126     String[] parts = name.split("\\/");
127     int depth = 0;
128     ScanNode parentNode = rootNode;
129 
130     // create any missing directory nodes.
131     for( int i = 0; i < parts.length - 1; i++ ) {
132       ScanNode currNode = parentNode.getChildMap().get(parts[i]);
133       if( currNode == null ) {
134         currNode = new ScanNode();
135         currNode.setResourceName("".equals(parentNode.getResourceName())?parts[i]:parentNode.getResourceName()+"/"+parts[i]);
136         currNode.setName(parts[i]);
137         currNode.setDirectory(true);
138         currNode.setFile(false);
139         parentNode.getChildMap().put(parts[i], currNode);
140       }
141       else {
142         currNode.setDirectory(true);
143       }
144       parentNode = currNode;
145     }
146 
147     // ASSERT: parentNode is now the parent node of this entry.
148 
149     // create the entry node if it is not already in the scan tree.
150     ScanNode currNode = parentNode.getChildMap().get(parts[parts.length-1]);
151     if( currNode == null ) {
152       currNode = new ScanNode();
153       currNode.setResourceName("".equals(parentNode.getResourceName())?parts[parts.length-1]:parentNode.getResourceName()+"/"+parts[parts.length-1]);
154       currNode.setName(parts[parts.length-1]);
155       currNode.setDirectory(isDirectory(name)?true:false);
156       currNode.setFile(isDirectory(name)?false:true);
157       parentNode.getChildMap().put(parts[parts.length-1], currNode);
158     }
159     else {
160       if( isDirectory(name) ) {
161         currNode.setDirectory(true);
162       }
163       else {
164         currNode.setFile(true);
165       }
166     }
167   }

169   private boolean isDirectory(String path) 
170   {
171     return path.endsWith("/");

174   public void scanDirectory( ScanNode scanNode, String path, Bundle bundle )
175     throws IOException
176   {
177     // The following excerpt from section 3.8.6 of the OSGi 4.2 Core Specification relates to bundle entry URLs:
178     //
179     //   The getPath method for a bundle entry URL must return an absolute path (a path that starts with '/') to a 
180     //   resource or entry in a bundle. For example, the URL returned from getEntry("myimages/test.gif") must have a 
181     //   path of /myimages/test.gif.
182     
183     // normalize the path
184     try {
185       path = new URI("/" + path + "/").normalize().getPath();
186     } catch( URISyntaxException e ) { 
187       throw new IllegalArgumentException(String.format("Given path '%s' has an invalid syntax.", path), e);
188     }
189     
190     if( ".".equals(path) ) { path = ""; }
191     Enumeration<URL> pathEnum = (Enumeration<URL>)bundle.findEntries(path, "*", true);
192     // make sure we have a normalized path.
193     if( pathEnum != null ) {
194       while( pathEnum.hasMoreElements() ) {
195         URL entry = pathEnum.nextElement();
196         if( entry.getPath().startsWith(path) ) {
197           String resourceName = entry.getPath().substring(path.length());
198           insertEntryNodes(scanNode, resourceName);
199         }
200       }
201     } 
202   }
