31  /**
32   * <p>This protocol scanner scans the protocol "jar:" and returns the root node of the scan.  The root node of the scan
33   * can be used to discover other nodes in the jar.</p>
34   *
35   * @author Christian Trimble
36   */
37  public class FileProtocolScanner
38    implements ProtocolScanner
39  {
40    public FileProtocolScanner()
41    {
42    }
43  
44    /**
45     * <p>Scans the entire directory at the specified url and adds all of the entries of that jar into the tree rooted at rootNode.  Missing
46     * directory nodes will be added.</p>
47     *
48     * @param rootNode the root node of the scan.
49     * @param jarUrl the url of the jar with entries to be added to the scan node.
50     */
51    public void scan( ScanNode rootNode, URL fileUrl )
52      throws Exception
53    {
54      // get the file object for the url object.
55      File rootDir = new File(fileUrl.toURI());
56  
57      // ASSERT: we now have a file handle for the file url.
58  
59      // add the children of this file to the root node.
60      for( File child : rootDir.listFiles() ) {
61        insertDirectoryNodes( rootNode, child );
62      }
63    }
64  
65    void insertDirectoryNodes( ScanNode parentNode, File file )
66    {
67      // create a node for this file, if it is missing.
68      ScanNode scanNode = parentNode.getChildMap().get(file.getName());
69      if( scanNode == null ) {
70        scanNode = new ScanNode();
71        scanNode.setResourceName("".equals(parentNode.getResourceName())?file.getName():parentNode.getResourceName()+"/"+file.getName());
72        scanNode.setName(file.getName());
73        scanNode.setDirectory(file.isDirectory());
74        scanNode.setFile(file.isFile());
75        parentNode.getChildMap().put(file.getName(), scanNode);
76      }
77  
78      if( file.isDirectory() ) {
79        for( File child : file.listFiles() ) {
80          insertDirectoryNodes( scanNode, child );
81        }
82      }
83    }
84  }
