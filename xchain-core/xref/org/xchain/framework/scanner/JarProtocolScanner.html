30  /**
31   * <p>This protocol scanner scans the protocol "jar:" and returns the root node of the scan.  The root node of the scan
32   * can be used to discover other nodes in the jar.</p>
33   *
34   * @author Christian Trimble
35   */
36  public class JarProtocolScanner
37    implements ProtocolScanner
38  {
39    /**
40     * <p>Scans the entire jar at the specified url and adds all of the entries of that jar into the tree rooted at rootNode.  Missing
41     * directory nodes will be added.</p>
42     *
43     * @param rootNode the root node of the scan.
44     * @param jarUrl the url of the jar with entries to be added to the scan node.
45     */
46    public void scan( ScanNode rootNode, URL jarUrl )
47      throws IOException
48    {
49      // NOTE: we are not closing this connection, because it was causing cached jars to be closed.  If this is causing problems with other
50      // jars that are not on the classpath, we may need to use JarInputStream to do this work.
51  
52      // get the jar from the url.
53      JarURLConnection conn = (JarURLConnection)jarUrl.openConnection();
54      JarFile jarFile = conn.getJarFile();
55  
56      // for each of the enties, get a jar entry and add it to the tree.
57      Enumeration<JarEntry> jarEntryEnum = jarFile.entries();
58      while( jarEntryEnum.hasMoreElements() ) {
59        JarEntry entry = jarEntryEnum.nextElement();
60        insertEntryNodes( rootNode, entry );
61      }
62    }
63  
64    /**
65     * <p>Adds an entry to the root node for a JarEntry.  This method is declared to be protected, so that test cases an interact with it.
66     * In general, this method should be treaded as private.</p>
67     *
68     * @param rootNode the root node of the scan.
69     * @param entry the jar entry to place under the root node.
70     */
71    void insertEntryNodes( ScanNode rootNode, JarEntry entry )
72    {
73      String name = entry.getName();
74      // TODO: verify that this works on windows.
75      String[] parts = name.split("\\/");
76      int depth = 0;
77      ScanNode parentNode = rootNode;
78  
79      // create any missing directory nodes.
80      for( int i = 0; i < parts.length - 1; i++ ) {
81        ScanNode currNode = parentNode.getChildMap().get(parts[i]);
82        if( currNode == null ) {
83          currNode = new ScanNode();
84          currNode.setResourceName("".equals(parentNode.getResourceName())?parts[i]:parentNode.getResourceName()+"/"+parts[i]);
85          currNode.setName(parts[i]);
86          currNode.setDirectory(true);
87          currNode.setFile(false);
88          parentNode.getChildMap().put(parts[i], currNode);
89        }
90        else {
91          currNode.setDirectory(true);
92        }
93        parentNode = currNode;
94      }
95  
96      // ASSERT: parentNode is now the parent node of this entry.
97  
98      // create the entry node if it is not already in the scan tree.
99      ScanNode currNode = parentNode.getChildMap().get(parts[parts.length-1]);
100     if( currNode == null ) {
101       currNode = new ScanNode();
102       currNode.setResourceName("".equals(parentNode.getResourceName())?parts[parts.length-1]:parentNode.getResourceName()+"/"+parts[parts.length-1]);
103       currNode.setName(parts[parts.length-1]);
104       currNode.setDirectory(entry.isDirectory()?true:false);
105       currNode.setFile(entry.isDirectory()?false:true);
106       parentNode.getChildMap().put(parts[parts.length-1], currNode);
107     }
108     else {
109       if( entry.isDirectory() ) {
110         currNode.setDirectory(true);
111       }
112       else {
113         currNode.setFile(true);
114       }
115     }
116   }
