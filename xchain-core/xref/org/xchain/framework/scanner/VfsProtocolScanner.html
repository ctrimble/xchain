16  package org.xchain.framework.scanner;
17  
18  import java.net.URL;
19  import org.jboss.virtual.VirtualFile;
20  import org.jboss.virtual.plugins.vfs.VirtualFileURLConnection;
21  
22  /**
23   * <p>A protocol handler for jboss 5.0.0 urls with protocols "vfszip", "vfsfile", and "vfsmemory".  This protocol handler interfaces with the VirtualFile objects exposed by
24   * jboss to build a tree of nodes to scan.</p>
25   *
26   * @author Christian Trimble
27   */
28  public class VfsProtocolScanner
29    implements ProtocolScanner
30  {
31    /**
32     * <p>Scans the VFS URL and adds the nodes in that url to the root node.</p>
33     *
34     * @param rootNode the root node of the resource tree.
35     * @param vfsUrl the root URL to scan.  The openConnection() method of this URL must return a connection of type VirtualFileURLConnection.
36     */
37    public void scan( ScanNode rootNode, URL vfsUrl )
38      throws Exception
39    {
40      // get the virtual file from the connection.
41      VirtualFileURLConnection vfConn = (VirtualFileURLConnection)vfsUrl.openConnection();
42      VirtualFile vFile = vfConn.getContent();
43  
44      // ASSERT: We now have the virtual file object.  We just need to recurse over it.
45  
46      for( VirtualFile child : vFile.getChildren() ) {
47        insertDirectoryNodes( rootNode, child );
48      }
49    }
50  
51    /**
52     * <p>Adds scan nodes to the parent node for the specified virtual file and all of its children.</p>
53     *
54     * @param parentNode the parent node of the 
55     */
56    void insertDirectoryNodes( ScanNode parentNode, VirtualFile vFile )
57      throws Exception
58    {
59      ScanNode scanNode = parentNode.getChildMap().get(vFile.getName());
60      if( scanNode == null ) {
61        scanNode = new ScanNode();
62        scanNode.setName(vFile.getName());
63        scanNode.setResourceName("".equals(parentNode.getResourceName())?vFile.getName():parentNode.getResourceName()+"/"+vFile.getName());
64        scanNode.setDirectory(!vFile.isLeaf());
65        scanNode.setFile(vFile.isLeaf());
66        parentNode.getChildMap().put(vFile.getName(), scanNode);
67      }
68  
69      if( !vFile.isLeaf() ) {
70        for( VirtualFile child : vFile.getChildren() ) {
71          insertDirectoryNodes( scanNode, child );
72        }
73      }
74    }
75  }
