26  public interface Permission extends Serializable {
27  
28    /**
29     * Implies is the mechanism by which we verify that an authenticated <code>Identity</code> can perform an action.
30     * 
31     * @param permission The operation being checked.
32     * @return <code>true</code> if the Identity is able to perform this operation, or a superset of this operation's functionality; <code>false</code> otherwise.

34    boolean implies(Permission permission);
