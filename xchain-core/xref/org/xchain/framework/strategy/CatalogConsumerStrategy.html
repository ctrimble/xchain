34  /**
35   * ConsumerStrategy implementation for Catalogs.
36   * 
37   * @author Devon Tackett
38   * @author Christian Trimble
39   * @author Josh Kennedy
40   *
41   * @see org.xchain.Catalog
42   */
43  public class CatalogConsumerStrategy implements ConsumerStrategy<Catalog, InputSource> {
44  	public static Logger log = LoggerFactory.getLogger( CatalogConsumerStrategy.class );
45  
46  	public Catalog consume(String systemId, SourceStrategy<InputSource> sourceStrategy, DependencyTracker tracker)
47  		throws Exception
48  	{
49  		// Get the input source
50  		InputSource inputSource = sourceStrategy.getSource(systemId);
51  		
52  	    // create the XMLReader.
53  	    XMLReader reader = XmlFactoryLifecycle.newXmlReader();
54  
55              reader.setErrorHandler( new FailingErrorHandler() );
56  
57      XChainDeclFilter sourceFilter = new XChainDeclFilter();
58      sourceFilter.setParent(reader);
59      sourceFilter.setErrorHandler( new FailingErrorHandler() );
60  
61  	    // create the jsl filter.
62  	    SaxTemplateHandler xmlFilter = new SaxTemplateHandler();
63  	    xmlFilter.setParent(sourceFilter);
64              xmlFilter.setErrorHandler( new FailingErrorHandler() );
65  
66  	    // create the digester, passing the jsl filter.
67  	    Digester digester = new Digester(xmlFilter);
68  
69  	    // set the digester onto the xml filter.
70  	    xmlFilter.setDigester(digester);
71  
72  	    // add the annotation rule set to the digester.
73  	    digester.addRuleSet(new AnnotationRuleSet(systemId));
74  
75  	    digester.setErrorHandler( new FailingErrorHandler() );
76  	    
77  	    // get the catalog object.
78  	    Catalog catalog = null;
79  
80              try {
81                catalog = (Catalog)digester.parse( inputSource );
82              }
83              catch( Exception e ) {
84                if( log.isErrorEnabled() ) {
85                  log.error("Could not create catalog for system id '"+inputSource.getSystemId()+"'.", e);
86                }
87                throw e;
88              }
89  		
90  		return catalog;
91  	}
92  
93    public static class FailingErrorHandler
94      implements ErrorHandler
95    {
96      public void warning(SAXParseException exception)
97        throws SAXException
98      {
99        if( log.isWarnEnabled() ) {
100         log.warn("SAXParseException thrown while loading catalog.", exception);
101       }
102     }
103 
104     public void error(SAXParseException exception)
105       throws SAXException
106     {
107       if( log.isErrorEnabled() ) {
108         log.error("SAXParseException thrown while loading catalog.", exception);
109       }
110       throw exception;
111     }
112 
113     public void fatalError(SAXParseException exception)
114       throws SAXException
115     {
116       if( log.isErrorEnabled() ) {
117         log.error("Fatal SAXParseException thrown while loading catalog.", exception);
118       }
119       throw exception;
120     }
121 
122   }
