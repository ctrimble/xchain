33  /**
34   * ConsumerStrategy implementation for Templates.
35   *
36   * @author Devon Tackett
37   * @author Christian Trimble
38   */
39  public class TemplatesConsumerStrategy implements ConsumerStrategy<SaxTemplates, InputSource> {
40  	
41  	public TemplatesConsumerStrategy() {
42  	}
43  
44  	public SaxTemplates consume(String systemId,
45  			SourceStrategy<InputSource> sourceStrategy,
46  			DependencyTracker tracker)
47  		throws Exception
48  	{   
49              // get the source.
50  	    InputSource source = sourceStrategy.getSource(systemId);
51  
52              // create the xml reader.
53              // NOTE: config option needed here to pick the xml parser used to load templates.
54              XMLReader reader = XmlFactoryLifecycle.newXmlReader();
55  
56              // create a templates handler.
57              SaxTemplatesHandler templatesHandler = XmlFactoryLifecycle.newTemplatesHandler();
58              reader.setContentHandler(templatesHandler);
59  
60              // parser the source.
61              reader.parse(source);
62  
63              // return the templats object.
64              return templatesHandler.getTemplates();
65  	}
