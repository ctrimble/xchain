58    public static Pattern getMethodPattern = null;
59    static {
60      try {
61        getMethodPattern = Pattern.compile("\\A(?:get|iterate)(.+)\\Z");
62      }
63      catch( PatternSyntaxException pse ) {
64        if( log.isErrorEnabled() ) {
65          log.error("Could not create the pattern for matching get methods.", pse);
66        }
67      }
68    }

70    /**
71     * Create a ClassPool which can access all the classes available to the given ClassLoader.
72     * 
73     * @param classLoader The ClassLoader to build the class path from.
74     * 
75     * @return A ClassPool which can access classes available to the given ClassLoader.
76     */
77    public static ClassPool createClassPool( ClassLoader classLoader )
78    {
79      ClassPool classPool = new ClassPool();
80      classPool.appendClassPath(new LoaderClassPath(classLoader));
81      return classPool;
82    }

84    /**
85     * Engineer the given command class.
86     * 
87     * @param classPool The ClassPool to build from.
88     * @param ctClass The command class to engineer.
89     * @return The engineered command class.
90     */
91    public static CtClass engineerCommand( ClassPool classPool, CtClass ctClass )
92      throws Exception
93    {
94      CtClass filterClass = getCtClass( classPool, "org.xchain.Filter");
95      CtClass locatableClass = getCtClass( classPool, "org.xchain.Locatable");
96      CtClass locatorClass = getCtClass( classPool, "org.xml.sax.Locator");
97      CtClass engineeredCommandClass = getCtClass( classPool, "org.xchain.EngineeredCommand");
98      CtClass qNameClass = getCtClass( classPool, "javax.xml.namespace.QName");
99      CtClass stringClass = getCtClass( classPool, "java.lang.String");
100     CtClass enumClass = getCtClass( classPool, "java.lang.Enum");
101     CtClass iteratorClass = getCtClass( classPool, "java.util.Iterator");
102 
103     // create the engineered class
104     CtClass engineeredClass = classPool.makeClass(getEngineeredCommandName(ctClass), ctClass);
105 
106       // if this class is not marked by the Engineered interface, then add it.
107       if( !engineeredClass.subtypeOf(engineeredCommandClass) ) {
108         engineeredClass.addInterface(engineeredCommandClass);
109       }
110       
111       // add a list of namespace prefix mappings to the command, so that they can be set during the execute and postProcess methods.
112       engineeredClass.addField(CtField.make("protected java.util.Map prefixMap = new java.util.HashMap();", engineeredClass));
113       CtMethod prefixMappingGetterMethod = CtNewMethod.make("public java.util.Map getPrefixMap() { return prefixMap; }", engineeredClass);
114       engineeredClass.addMethod(prefixMappingGetterMethod);
115 
116       engineeredClass.addField(CtField.make("protected java.util.Map attributeMap = new java.util.HashMap();", engineeredClass));
117       engineeredClass.addMethod(CtNewMethod.make("public java.util.Map getAttributeMap() { return attributeMap; }", engineeredClass));
118 
119       engineeredClass.addField(CtField.make("protected javax.xml.namespace.QName qName = null;", engineeredClass));
120       engineeredClass.addMethod(CtNewMethod.make("public void setQName( javax.xml.namespace.QName qName ) { this.qName = qName; }", engineeredClass));
121       engineeredClass.addMethod(CtNewMethod.make("public javax.xml.namespace.QName getQName() { return this.qName; }", engineeredClass));
122 
123       engineeredClass.addField(CtField.make("protected String systemId = null;", engineeredClass));
124       engineeredClass.addMethod(CtNewMethod.make("public void setSystemId( String systemId ) { this.systemId = systemId; }", engineeredClass));
125       engineeredClass.addMethod(CtNewMethod.make("public String getSystemId() { return this.systemId; }", engineeredClass));
126 
127       engineeredClass.addMethod(CtNewMethod.make("public boolean isRegistered() { return this.qName != null && this.systemId != null; }", engineeredClass));
128 
129       // add the locatable interface if it is missing.
130       if( !engineeredClass.subtypeOf(locatableClass) ) {
131         engineeredClass.addInterface(locatableClass);
132       }
133 
134       // add or override locatable methods.
135       CtMethod setLocatorMethod = getMethod( engineeredClass, "setLocator", new CtClass[] { locatorClass } );
136       CtMethod getLocatorMethod = getMethod( engineeredClass, "getLocator", new CtClass[] {} );
137       CtField locatorField = getField( engineeredClass, "locator" );
138 
139       // make sure the method getLocator() has the correct return type.
140       if( getLocatorMethod != null && !getLocatorMethod.getReturnType().subtypeOf(locatorClass) ) {
141         throw new Exception("The class '"+engineeredClass.getName()+"' is not compatable with the interface org.xchain.Locatable.  The return type of getLocator() is incorrect.");
142       }
143 
144       // make sure the method setLocator(Locator) has the correct return type.
145       if( setLocatorMethod != null && !setLocatorMethod.getReturnType().subtypeOf(CtClass.voidType) ) {
146         throw new Exception("The class '"+engineeredClass.getName()+"' is not compatable with the interface org.xchain.Locatable.  The return type of setLocator( Locator locator ) is incorrect.");
147       }
148 
149       // if we need to implement either the setLocator(Locator) method or the getLocator() methods, then we need to check the locator fields type.
150       if( (setLocatorMethod == null || getLocatorMethod == null) && locatorField != null && !locatorField.getType().subtypeOf(locatorClass) ) {
151         throw new Exception("The class '"+engineeredClass.getName()+"' has a locator field that is not of the type org.xml.sax.Locator.");
152       }
153 
154       if( ( setLocatorMethod == null || getLocatorMethod == null ) && locatorField == null ) {
155         engineeredClass.addField(CtField.make("protected org.xml.sax.Locator locator = null;", engineeredClass));
156       }
157 
158       if( setLocatorMethod == null ) {
159         engineeredClass.addMethod(CtNewMethod.make("public void setLocator( org.xml.sax.Locator locator ) { this.locator = locator; }", engineeredClass));
160       }
161       if( getLocatorMethod == null ) {
162         engineeredClass.addMethod(CtNewMethod.make("public org.xml.sax.Locator getLocator() { return this.locator; }", engineeredClass));
163       }
164       
165       // Keep track of all the attributes for this engineered command.
166       Set<QName> attributeSet = new HashSet<QName>();
167       Map<QName, AttributeType> attributeTypeMap = new HashMap<QName, AttributeType>();
168       Map<QName, Boolean> attributeRequiredMap = new HashMap<QName, Boolean>();
169       Map<QName, CtClass> attributeJavaTypeMap = new HashMap<QName, CtClass>();
170 
171       // add attribute fields to abstract methods.
172       for( CtMethod ctMethod : ctClass.getMethods() ) {
173         // if the ctMethod has the attribute annotations and is abstract, then add the needed methods.
174         if( hasAnnotation( ctMethod, Attribute.class ) ) {
175           // variables from the annotation.
176           String attributeLocalName = (String)getAnnotationValue( ctMethod, Attribute.class, "localName" );
177           String attributeUri = (String)getAnnotationValue( ctMethod, Attribute.class, "namespaceUri" );
178           String defaultValue = (String)getAnnotationValue( ctMethod, Attribute.class, "defaultValue" );
179           Boolean required = (Boolean)getAnnotationValue( ctMethod, Attribute.class, "required" );
180 
181           // variables describing the return type.
182           boolean isPrimative = ctMethod.getReturnType().isPrimitive();
183           boolean isVoid = ctMethod.getReturnType().equals(CtClass.voidType);
184           boolean isBoolean = ctMethod.getReturnType().equals(CtClass.booleanType);
185           boolean isNumber = ctMethod.getReturnType().equals(CtClass.byteType) ||
186                              ctMethod.getReturnType().equals(CtClass.charType) ||
187                              ctMethod.getReturnType().equals(CtClass.doubleType) ||
188                              ctMethod.getReturnType().equals(CtClass.floatType) ||
189                              ctMethod.getReturnType().equals(CtClass.intType) ||
190                              ctMethod.getReturnType().equals(CtClass.longType) ||
191                              ctMethod.getReturnType().equals(CtClass.shortType);
192           boolean isTypeSpecified = ctMethod.getParameterTypes().length == 2;
193           // the type that will be used with ConvertUtils and JXPathContext
194           String typeExpression = isTypeSpecified ? "$2" : "$type";
195           boolean isIterateMethod = ctMethod.getName().startsWith("iterate") && ctMethod.getReturnType().subtypeOf(iteratorClass);
196 
197           // add a default value of primative types.
198           if( defaultValue == null && isPrimative ) {
199             if( isBoolean ) { defaultValue = "false"; }
200             else if( isNumber ) { defaultValue = "("+ctMethod.getReturnType().getSimpleName()+")0;"; }
201             else if( isVoid ) { /* void does not get a default value */ }
202             else {
203               throw new IllegalStateException("The engineering code does not support the primative type '"+ctMethod.getReturnType().getSimpleName());
204             }
205           }
206           //
207           // Now we are going to engineer a method following this pattern:
208           //
209           // {
210           //   // The name for this attribute.
211           //   javax.xml.namespace.QName attributeName = new javax.xml.namespace.QName(ATTRIBUTE_NAMESPACE_URI, ATTRIBUTE_LOCAL_NAME);
212           //
213           //   // The attribute value defined on this element.
214           //   String attributeValue = (String)getAttributeMap().get(attributeName);
215           //
216           //   // Do we need use the default?
217           //   boolean doDefault = attributeValue == null;
218           //
219           //   // The map that will hold the original mappings.
220           //   java.util.Map originalPrefixMapping = null;
221           //
222           //   // NOTE: If the attribute is required, but the attribute is not present, we need to throw an exception.
223           //
224           //   // ASSERT: at this point, we are going to have a value to return, or fail with a type conversion exception.
225           //   ATTRIBUTE_TYPE result;
226           //   try {
227           //     // define the prefix mappings for this commands element.
228           //     org.xchain.framework.lifecycle.Execution.definePrefixMappings($1, this);
229           //
230           //     // if we need a default and we have a default, then do the default mappings.
231           //     if( doDefault ) {
232           //       originalPrefixMapping = new java.util.HashMap();
233           //
234           //       // record all of the current mappings for the default prefixes.
235           //       originalPrefixMapping.put(DEFAULT_MAPPING_PREFIX, $1.getNamespaceURI(DEFAULT_MAPPING_PREFIX));
236           //       ...
237           //       // register the default namespace mappings.
238           //       $1.registerNamespace(DEFAULT_MAPPING_PREFIX, DEFAULT_MAPPING_URI);
239           //       ...
240           //       attributeValue = defaultValue;
241           //     }
242           //       
243           //     // Do the AttributeType specific lookup.  This changes per attribute type.
244           //   }
245           //   catch( org.apache.commons.jxpath.JXPathException jxpe ) {
246           //     Throwable cause = jxpe.getCause();
247           //     if( cause == null ) {
248           //       throw jxpe;
249           //     }
250           //     else if( cause instanceof RuntimeException ) {
251           //       throw (RuntimeException)cause;
252           //     }
253           //     else if( cause instanceof THROWN_EXCEPTION ) {
254           //       throw (THROWN_EXCEPTION)cause;
255           //     }
256           //     ..
257           //   }
258           //   finally {
259           //     // if there are default mappings, include code to undo the mappings.
260           //     if( doDefault && originalPrefixMapping != null ) {
261           //       // there is one of there for each mapping.
262           //       $1.registerNamespace(DEFAULT_MAPPING_PREFIX, (String)originalPrefixMapping.get(DEFAULT_MAPPING_PREFIX));
263           //       ...
264           //     }
265           //     // undefine the prefix mappings for this commands element.
266           //     org.xchain.framework.lifecycle.Execution.undefinePrefixMappings($1, this);  }
267           //   }
268           //   return value;
269 
270           QName attributeQName = new QName(attributeUri, attributeLocalName);
271           AttributeType attributeType = (AttributeType)getAnnotationValue( ctMethod, Attribute.class, "type" );
272           CtClass attributeJavaType = ctMethod.getReturnType();
273           // Add the attribute to the set of known attributes.
274           attributeSet.add(attributeQName);
275           attributeTypeMap.put(attributeQName, attributeType);
276           attributeJavaTypeMap.put(attributeQName, attributeJavaType);
277           attributeRequiredMap.put(attributeQName, required);
278           PrefixMapping[] defaultPrefixMappings = (PrefixMapping[])getAnnotationValue( ctMethod, Attribute.class, "defaultPrefixMappings" );
279           if( Modifier.isAbstract(ctMethod.getModifiers()) ) {
280           CtMethod engineeredMethod = CtNewMethod.copy( ctMethod, engineeredClass, null );
281 
282           // Start making the method body.
283           StringBuffer methodBody = new StringBuffer();
284           methodBody.append("{\n");
285           methodBody.append("  // The name for this attribute.\n");
286           methodBody.append("  javax.xml.namespace.QName attributeName = new javax.xml.namespace.QName(")
287                     .append(stringConstant(attributeUri)).append(", ").append(stringConstant(attributeLocalName)).append(");\n");
288           methodBody.append("\n");
289           methodBody.append("  // The attribute value defined on this element.\n");
290           methodBody.append("  String attributeValue = (String)getAttributeMap().get(attributeName);\n");
291           methodBody.append("\n");
292           methodBody.append("  // Do we need use the default?\n");
293           methodBody.append("  boolean doDefault = attributeValue == null;\n");
294           methodBody.append("\n");
295           methodBody.append("  // The map that will hold the original mappings.\n");
296           methodBody.append("  java.util.Map originalPrefixMapping = null;\n");
297           methodBody.append("\n");
298 
299           // TODO: add code here to handle required.
300           // if the attributeValue is null, we are not in a primative, the default is "", and the attribute is not required, return null.
301           if( !isPrimative && "".equals(defaultValue)) {
302             methodBody.append("  if( attributeValue == null ) {\n");
303             methodBody.append("    return null;\n");
304             methodBody.append("  }\n");
305             methodBody.append("\n");
306           }
307 
308           methodBody.append("  // ASSERT: at this point, we are going to have a value to return, or fail with a type conversion exception.\n");
309           methodBody.append("\n");
310           methodBody.append("  // This is the variable to hold the result.\n");
311           methodBody.append("  ").append(ctMethod.getReturnType().getName()).append(" result;\n");
312           methodBody.append("\n");
313           methodBody.append("  try {\n");
314           methodBody.append("    // define the prefix mappings for this commands element.\n");
315           methodBody.append("    org.xchain.framework.lifecycle.Execution.definePrefixMappings($1, this);\n");
316           methodBody.append("\n");
317           methodBody.append("    // if we need a default and we have a default, then do the default mappings.\n");
318           methodBody.append("    if( doDefault ) {\n");
319           methodBody.append("      originalPrefixMapping = new java.util.HashMap();\n");
320           methodBody.append("\n");
321           methodBody.append("      // record all of the current mappings for the default prefixes.\n");
322           for( PrefixMapping mapping : defaultPrefixMappings ) {
323             String escapedPrefix = stringConstant(mapping.prefix());
324             methodBody.append("      originalPrefixMapping.put(").append(escapedPrefix).append(", $1.getNamespaceURI(").append(escapedPrefix).append("));\n");
325           }
326           methodBody.append("\n");
327           methodBody.append("      // register the default namespace mappings.\n");
328           for( PrefixMapping mapping : defaultPrefixMappings ) {
329             String escapedPrefix = stringConstant(mapping.prefix());
330             String escapedUri = stringConstant(mapping.uri());
331             methodBody.append("      $1.registerNamespace(").append(escapedPrefix).append(", ").append(escapedUri).append(");\n");
332           }
333           methodBody.append("\n");
334           if( !"".equals(defaultValue) ) {
335             methodBody.append("      attributeValue = ").append(stringConstant(defaultValue)).append(";\n");
336           }
337           else if( isNumber ) {
338             methodBody.append("      attributeValue = \"0\";\n");
339           }
340           else if( isBoolean ) {
341             methodBody.append("      attributeValue = \"false\";\n");
342           }
343           methodBody.append("    }\n");
344           methodBody.append("\n");
345 
346           // ASSERT: The top of the method definition is now done.  The value of the attribute has now been looked up, defaulted, or the method would have caused an exception.
347 
348           // It is now time to add the logic for the AttriuteType.
349           switch( attributeType ) {
350             // 
351             // AttributeType.JXPATH_VALUE attributes follow one of two patterns... 
352             // For methods that start with "iterate" and have a type of java.util.Iterator, the following code is used:
353             //   result = ($r)$1.iterate(attributeValue);
354             //
355             // For all other methods, the following code is used:
356             //   if( !TYPE.isAssignableFrom(REQUESTED_TYPE) {
357             //     throw new IllegalArgumentException("Can not cast REQUESTED_TYPE into TYPE.");
358             //   }
359             //   result = ($r)$1.getValue(attributeValue, TYPE);
360             case JXPATH_VALUE:
361               if( isIterateMethod ) {
362                 methodBody.append("    result = ($r)$1.iterate(attributeValue);\n");
363               }
364               else {
365                 methodBody.append("    result = ($r)$1.getValue(attributeValue, ").append(typeExpression).append(");\n");
366               }
367               break;
368             case JXPATH_SELECT_NODES:
369               if( isIterateMethod ) {
370                 methodBody.append("    result = ($r)$1.selectNodes(attributeValue).iterator();\n");
371               }
372               else {
373                 methodBody.append("    return ($r)$1.selectNodes(attributeValue);\n");
374               }
375               break;
376             case JXPATH_SELECT_SINGLE_NODE:
377               methodBody.append("    return ($r)$1.selectSingleNode(attributeValue);\n");
378               break;
379             case JXPATH_POINTER:
380               methodBody.append("    result = ($r)$1.getPointer(attributeValue);\n");
381               break;
382             case JXPATH_ITERATE_POINTERS:
383               methodBody.append("    result = ($r)$1.iteratePointers(attributeValue);\n");
384               break;
385             case QNAME:
386               if( engineeredMethod.getReturnType().subtypeOf(qNameClass) ) {
387                 methodBody.append("    result = org.xchain.framework.util.JXPathContextUtil.stringToQName($1, attributeValue);\n");
388               }
389               else if( engineeredMethod.getReturnType().subtypeOf(stringClass) ) {
390                 methodBody.append("    result = org.xchain.framework.util.JXPathContextUtil.stringToQNameString($1, attributeValue);\n");
391               }
392               else {
393                 throw new RuntimeException("QName attributes do not support the return type '"+engineeredMethod.getReturnType()+"'.");
394               }
395               break;
396             case LITERAL:
397               methodBody.append("    result = ($r)org.xchain.framework.util.JXPathContextUtil.convert((Object)attributeValue, ").append(typeExpression).append(");\n");
398               break;
399             case ATTRIBUTE_VALUE_TEMPLATE:
400               methodBody.append("    result = ($r)org.xchain.framework.util.JXPathContextUtil.convert((Object)org.xchain.framework.util.AttributesUtil.evaluateAttributeValueTemplate($1, attributeValue), ")
401                         .append(typeExpression).append(");\n");
402               break;
403             default:
404               throw new RuntimeException("Unknown attribute type encountered.");
405           }
406           methodBody.append("  }\n");
407 
408           // ASSERT: The method body now has a complete try block.
409 
410           // JXPath wraps exceptions coming from functions in JXPathException's, we need to unwrap those, so that the caller get the exception that they expect.
411           methodBody.append("  catch( org.apache.commons.jxpath.JXPathException jxpe ) {\n");
412           methodBody.append("    Throwable cause = jxpe.getCause();\n");
413           methodBody.append("    // if there is no cause, then we cannot unwrap the exception.\n");
414           methodBody.append("    if( cause == null ) {\n");
415           methodBody.append("      throw jxpe;\n");
416           methodBody.append("    }\n");
417           methodBody.append("    // if this is a runtime exception, then we can throw it as a runtime exception.\n");
418           methodBody.append("    else if( cause instanceof RuntimeException ) {\n");
419           methodBody.append("      throw (RuntimeException)cause;\n");
420           methodBody.append("    }\n");
421           // for each exception that is declared to be thrown, we need to test the runnable and try to throw as that type.
422           for( CtClass thrownExceptionClass : ctMethod.getExceptionTypes() ) {
423             methodBody.append("    else if( cause instanceof ").append(thrownExceptionClass.getName()).append(" ) {\n");
424             methodBody.append("      throw (").append(thrownExceptionClass.getName()).append(")cause;\n");
425             methodBody.append("    }\n");
426           }
427           methodBody.append("    // we could not cast the cause, so leave the exception wrapped.\n");
428           methodBody.append("    else {\n");
429           methodBody.append("      throw jxpe;\n");
430           methodBody.append("    }\n");
431           methodBody.append("  }\n");
432 
433           // ASSERT: Any exceptions thrown while evaluating an attribute have been unwrapped if they could.
434 
435           methodBody.append("  finally {\n");
436           methodBody.append("    // if there are default mappings, include code to undo the mappings.\n");
437           methodBody.append("    if( doDefault && originalPrefixMapping != null ) {\n");
438           methodBody.append("      // there is one of there for each mapping.\n");
439           for( PrefixMapping mapping : defaultPrefixMappings ) {
440             String escapedPrefix = stringConstant(mapping.prefix());
441             methodBody.append("      $1.registerNamespace(").append(escapedPrefix).append(", (String)originalPrefixMapping.get(").append(escapedPrefix).append("));\n");
442           }
443           methodBody.append("    }\n");
444           methodBody.append("    // undefine the prefix mappings for this commands element.\n");
445           methodBody.append("    org.xchain.framework.lifecycle.Execution.undefinePrefixMappings($1, this);\n");
446           methodBody.append("  }\n");
447           methodBody.append("  return result;\n");
448           methodBody.append("}\n");
449             //System.out.println(methodBody.toString());
450             //Thread.currentThread().sleep(1000);
451             engineeredMethod.setBody(methodBody.toString());
452             engineeredMethod.setModifiers( Modifier.clear( ctMethod.getModifiers(), Modifier.ABSTRACT ) );
453             engineeredClass.addMethod(engineeredMethod);
454           }
455 
456           // test to see if the class has a "has" method.
457           Matcher matcher = getMethodPattern.matcher(ctMethod.getName());
458           if( matcher.find() ) {
459             String hasMethodName = "has"+matcher.group(1);
460             try {
461               CtMethod hasMethod = ctClass.getMethod( hasMethodName, "()Z");
462               if( Modifier.isAbstract(hasMethod.getModifiers()) ) {
463                 CtMethod engineeredHasMethod = CtNewMethod.copy( hasMethod, engineeredClass, null );
464                 // TODO: should we return true if there is a default value?
465                 engineeredHasMethod.setBody(
466                   "{"+
467                     "return getAttributeMap().containsKey(new javax.xml.namespace.QName(\""+attributeUri+"\", \""+attributeLocalName+"\"));"+
468                   "}"
469                 );
470                 engineeredHasMethod.setModifiers( Modifier.clear( hasMethod.getModifiers(), Modifier.ABSTRACT ) );
471                 engineeredClass.addMethod(engineeredHasMethod);
472               }
473             }
474             catch( NotFoundException nfe ) {
475               // do nothing, we do not need this method.
476             }
477           }
478         }  
479       }
480 
481       // Add all static fields and value.
482       // Add the static attributes.
483       engineeredClass.addField(CtField.make("protected static java.util.Set attributeSet = new java.util.HashSet();", engineeredClass));
484       engineeredClass.addField(CtField.make("protected static java.util.Map attributeDetailMap = new java.util.HashMap();", engineeredClass));
485 
486       // Add the class initializer for the static attributes.
487       StringBuilder ib = new StringBuilder();
488       ib.append("{");
489       ib.append("javax.xml.namespace.QName attributeQName = null;");
490       ib.append("org.xchain.AttributeDetail attributeDetail = null;");
491       for(QName attributeName : attributeSet) {
492         AttributeType attributeType = attributeTypeMap.get(attributeName);
493         CtClass attributeJavaType = attributeJavaTypeMap.get(attributeName);
494         ib.append("attributeQName = new javax.xml.namespace.QName(\"" + attributeName.getNamespaceURI() + "\", \"" + attributeName.getLocalPart() + "\");");
495         ib.append("attributeSet.add(attributeQName);");
496         ib.append("attributeDetail = new org.xchain.AttributeDetail(attributeQName, org.xchain.annotations.AttributeType."+attributeType.name()+", "+attributeJavaType.getName()+".class, "+attributeRequiredMap.get(attributeName)+");");
