39  public class TestSessionCommand
40  {
41    public static final QName TEST_DEFAULT_SESSION = new QName("test-default-session");
42    public static final QName TEST_DEFINED_NAMED_SESSION = new QName("", "test-defined-named-session");
43    public static final QName TEST_UNDEFINED_NAMED_SESSION = new QName("", "test-undefined-named-session");
44    public static final QName TEST_MULTIPLE_DEFINED_NAMED_SESSIONS = new QName("", "test-multiple-defined-named-sessions");
45  
46    public static String SUCCESS_VARIABLE = "test-success";
47  
48    protected JXPathContext context = null;
49    protected Catalog catalog = null;
50  
51    @Before public void setUp()
52      throws Exception
53    {
54      Lifecycle.startLifecycle();
55  
56      // get the catalog.
57      catalog = CatalogFactory.getInstance().getCatalog("resource://context-class-loader/org/xchain/namespaces/hibernate/test-session-command.xchain");
58  
59      // create the context.
60      context = JXPathContext.newContext(new Object());
61    }
62  
63    @After public void tearDown()
64      throws Exception
65    {
66      context = null;
67      catalog = null;
68  
69      Lifecycle.stopLifecycle();
70    }
71  
72    @Test public void testDefaultSession()
73      throws Exception
74    {
75      Command command = catalog.getCommand(TEST_DEFAULT_SESSION);
76      command.execute(context);
77      assertTrue("Session creation failed.", context.getVariables().isDeclaredVariable(SUCCESS_VARIABLE) );
78    }
79  
80    @Test public void testNamedSession()
81      throws Exception
82    {
83      Command command = catalog.getCommand(TEST_DEFINED_NAMED_SESSION);
84      command.execute(context);
85      assertTrue("Session creation failed.", context.getVariables().isDeclaredVariable(SUCCESS_VARIABLE) );
86    }
87  
88    @Test public void testUndefinedNamedSession()
89      throws Exception
90    {
91      Command command = catalog.getCommand(TEST_UNDEFINED_NAMED_SESSION);
92  
93      try {
94        command.execute(context);
95      }
96      catch( Throwable t ) {
97        assertTrue("The session command threw the type '"+t.getClass().getName()+"' when asked to create an undefined named session.", t instanceof ExecutionException);
98        assertNotNull("The session command threw an exception with a null cause.", t.getCause());
99        assertTrue("The session command threw an exception with a cause of type '"+t.getCause().getClass().getName()+"' when asked to create an undefined session.", t.getCause() instanceof SessionFactoryNotFoundException);
100       return;
101     }
102     fail("The session command created a command for an undefined named session.");
103   }
104 
105   @Test public void testMultipleDefinedNamedSession()
106     throws Exception
107   {
108     Command command = catalog.getCommand(TEST_MULTIPLE_DEFINED_NAMED_SESSIONS);
109     command.execute(context);
110     assertTrue("Session creation failed.", context.getVariables().isDeclaredVariable(SUCCESS_VARIABLE) );
111   }
