18  import java.util.Properties;
19  import javax.naming.InitialContext;
20  import javax.naming.NamingException;
21  import javax.sql.DataSource;
22  import org.hibernate.HibernateException;
23  import org.hibernate.cfg.Environment;
24  import org.hibernate.connection.ConnectionProvider;
25  import org.hibernate.util.NamingHelper;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import java.sql.Connection;

31  /**
32   * A data source connection provider that will rebind the data source if it fails to create a connection.  When using the 'hibernate.connection.datasource' property, XChains will use this
33   * connection provider, unless the property 'hibernate.connection.provider_class' has been specified in the configuration.
34   *
35   * @author Christian Trimble
36   * @author Josh Kennedy
37   */
38  public class RebindingDataSourceConnectionProvider
39    implements ConnectionProvider
40  {
41    /** The log for this data source provider. */
42    private static Logger log = LoggerFactory.getLogger(RebindingDataSourceConnectionProvider.class);
43  
44    /** The jndi name of the data source. */
45    private String jndiName = null;
46  
47    /** The user name for creating connections. */
48    private String userName = null;
49  
50    /** The password for creating connections. */
51    private String password = null;
52  
53    /** The initial context for looking up data sources. */
54    private InitialContext initialContext = null;
55  
56    /** The current data source for creating connections. */
57    private DataSource dataSource = null;
58  
59    /**
60     * Configures this connection provider with the specified properties.  This method will only fail with a hibernate exception if the 'hibernate.connection.datasource' property is not specified,
61     * or if the JNDI initial context cannot be created.  Otherwise, this method will log any problems with the configuration, or looking up the data source and return normally.
62     *
63     * @throws HibernateException if the property 'hibernate.connection.datasource' property is not specified, or if the JNDI initial context cannot be created.
64     */
65    public void configure(Properties properties)
66      throws HibernateException
67    {
68      jndiName = properties.getProperty( Environment.DATASOURCE );
69      userName = properties.getProperty( Environment.USER );
70      password = properties.getProperty( Environment.PASS );
71  
72      if( jndiName == null ) {
73        if( log.isErrorEnabled() ) {
74          log.error("The JNDI name of the JDBC data source was not specified.  Please set the hibernate property '"+Environment.DATASOURCE+"'.");
75        }
76        throw new HibernateException("The JNDI name of the JDBC data source was not specified.  Please set the hibernate property '"+Environment.DATASOURCE+"'.");
77      }
78  
79      if( userName != null && password == null && log.isWarnEnabled() ) {
80        log.warn("A JDBC user name and password will not be used because the hibernate property '"+Environment.PASS+"' is not set.");
81      }
82  
83      if( userName == null && password != null && log.isWarnEnabled() ) {
84        log.warn("A JDBC user name and password will not be used because the hibernate property '"+Environment.USER+"' is not set.");
85      }
86  
87      try {
88        initialContext = NamingHelper.getInitialContext(properties);
89      }
90      catch( NamingException ne ) {
91        throw new HibernateException("Could not create initial context for looking up datasource.", ne);
92      }
93  
94      try {
95        initializeDataSource();
96      }
97      catch( SQLException sqle ) {
98        // this excetion has been logged.  let startup continue, as the application will start responding once the configuration issues are resolved.
99      }
100   }
101 
102   /**
103    * Initialized, or reinitializes, the data source for this connection provider.  This method logs any errors that prevented the initialization of the data source and will only update
104    * the datasource reference if a new datasource could be obtained.
105    *
106    * @throws SQLException if the data source could not be loaded from JNDI for any reason.
107    */
108   private void initializeDataSource()
109     throws SQLException
110   {
111     Object jndiObject = null;
112 
113     // Look up the object bound to the jndi name.
114     try {
115       jndiObject = initialContext.lookup(jndiName);
116     }
117     catch( NamingException ne ) {
118       if( log.isErrorEnabled() ) {
119         log.error("A JNDI exception was thrown while looking up the JDBC data source at '"+jndiName+"'.", ne);
120       }
121       throw new SQLException("A JNDI exception was thrown while looking up the JDBC data source at '"+jndiName+"'.");
122     }
123 
124     // If we did not find an object, bail out.
125     if( jndiObject == null ) {
126       if( log.isErrorEnabled() ) {
127         log.error("There is not a javax.sql.DataSource bound to the JNDI name '"+jndiName+"'.  This JNDI name is not bound to an object.");
128       }
129       throw new SQLException("There is not a javax.sql.DataSource bound to the JNDI name '"+jndiName+"'.  This JNDI name is not bound to an object.");
130     }
131 
132     // if the object is not a data source, bail out.
133     if( !(jndiObject instanceof DataSource) ) {
134       if( log.isErrorEnabled() ) {
135         log.error("There is not a javax.sql.DataSource bound to the JNDI name '"+jndiName+"'.  This JNDI name is currently bound to an object of type '"+jndiObject.getClass().getName()+"'.");
136       }
137       throw new SQLException("There is not a javax.sql.DataSource bound to the JNDI name '"+jndiName+"'.  This JNDI name is currently bound to an object of type '"+jndiObject.getClass().getName()+"'.");
138     }
139 
140     // We are ready to set this data source.
141     dataSource = (DataSource)jndiObject;
142   }

144   /**
145    * Returns a connection from the currently configured JDBC data source.
146    *
147    * @throws SQLException if the data source is null, or if the data source throws an exception while creating the connection.
148    * @return a connection from the currently configured JDBC data source.
149    */
150   private Connection getConnectionFromDataSource()
151     throws SQLException
152   {
153     if( dataSource == null ) {
154       throw new SQLException("A JDBC data source could not be found at JNDI name '"+jndiName+"'.");
155     }
156 
157     if( userName != null && password != null ) {
158       return dataSource.getConnection(userName, password);
159     }
160     else {
161       return dataSource.getConnection();
162     }
163   }
164 
165   /**
166    * Gets a connection from the datasource bound to jndi.  If getting the connection fails, this method will attempt to rebind the datasource once, then fail if rebinding fails or the newly bound
167    * DataSource fails.  This method is synchronized, so that not more than one thread attempts to rebind the datasource at a time.  This should not cause a major performance hit, since the datasource
168    * pool will most likely be synchronized as well.
169    *
170    * @throws SQLException if a fresh copy of the datasource from JNDI caused an exception to be thrown.
171    * @return a connection to the data source, creating a new data source if required.
172    */
173   public synchronized Connection getConnection()
174     throws SQLException
175   {
176     try {
177       return getConnectionFromDataSource();
178     }
179     catch( Exception exception ) { 
180       if( log.isWarnEnabled() ) {
181         log.warn("Attempting to rebind to the jndi datasource due to an exception.", exception);
182       }
183       initializeDataSource();
184       return getConnectionFromDataSource();
185     }
186   }

192    * @throws SQLException if an exception is thrown while closing the connection.
193    */
194   public void closeConnection( Connection connection )
195     throws SQLException
196   {
197     connection.close();
198   }

200   /**
201    * Closes this connection provider.
202    */
203   public void close()
204   {
205     initialContext = null;
206     dataSource = null;
207     jndiName = null;
208     userName = null;
209     password = null;
210   }
