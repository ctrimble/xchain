33  /**
34   * <p>NOTE: The hibernate session command is currently broken.  This implementation will create duplicate sessions
35   * when used with the HibernateLifecycle.getCurrentSession() function.  This tag will be removed in the 0.4.0 version
36   * of XChains.</p>
37   *
38   * <p>The <code>session</code> command creates a new Hibernate session and stores it into the context.  
39   * The location of the session in the context is determined by the <code>result</code> attribute.</p>
40   * 
41   * <p>Once the <code>session</code> command is complete, the Hibernate session will be closed.</p>
42   * 
43   * <code class="source">
44   * &lt;xchain:session xmlns:xchain="http://www.xchain.org/hibernate/1.0"&gt;
45   *  ...
46   * &lt;/xchain:session&gt;
47   * </code>
48   *
49   * @author Mike Moulton
50   * @author Devon Tackett
51   * @author Christian Trimble
52   * @author John Trimble
53   * @author Josh Kennedy
54   */
55  @Element(localName="session")
56  public abstract class SessionCommand
57    extends ChainImpl
58  {
59    public static Logger log = LoggerFactory.getLogger(SessionCommand.class);
60  
61    /**
62     * Where the session is stored. 
63     */
64    @Attribute(localName="result",
65               type=AttributeType.QNAME,
66               defaultValue= "{" + Constants.URI + "}" + Constants.SESSION)             
67    public abstract QName getResult( JXPathContext context );
68  
69    @Attribute(localName="name",
70               type=AttributeType.QNAME,
71               defaultValue= "{http://www.xchain.org/hibernate}session-factory")
72    public abstract QName getName( JXPathContext context );
73  
74    public boolean execute( JXPathContext context )
75      throws Exception
76    {
77      if( log.isErrorEnabled() ) {
78        log.error("This tag has been removed in XChains 0.3.0.  Remove this tag and replace references to the $hibernate:session varaible with a call to hibernate:current-session()");
79      }
80      throw new UnsupportedOperationException("This tag has been removed in XChains 0.3.0.  Remove this tag and replace references to the $hibernate:session varaible with a call to hibernate:current-session()");
81    }
82  
83  }
